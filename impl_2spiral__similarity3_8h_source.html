44  static auto areParamsValid(
45  Eigen::Vector<Scalar, kNumParams> const& non_zero_quat)
46  -> sophus::Expected<Success> {
47  static const Scalar kThr = kEpsilon<Scalar> * kEpsilon<Scalar>;
48  const Scalar squared_norm = non_zero_quat.squaredNorm();
49  using std::abs;
50  if (!(squared_norm > kThr || squared_norm < 1.0 / kThr)) {
51  return SOPHUS_UNEXPECTED(
52  "complex number ({}, {}) is too large or too small.\n"
53  "Squared norm: {}, thr: {}",
54  non_zero_quat[0],
55  non_zero_quat[1],
56  squared_norm,
57  kThr);
58  }
59  return sophus::Expected<Success>{};
60  }

64  static auto exp(Eigen::Vector<Scalar, kDof> const& omega_logscale)
65  -> Eigen::Vector<Scalar, kNumParams> {
66  using std::exp;
67  using std::max;
68  using std::min;
69  using std::sqrt;
70 
71  Eigen::Vector3<Scalar> const vec_omega = omega_logscale.template head<3>();
72  Scalar scale = exp(omega_logscale[3]);
73  // Ensure that scale-factor contraint is always valid
74  scale = max(scale, kEpsilonPlus<Scalar>);
75  scale = min(scale, Scalar(1.) / kEpsilonPlus<Scalar>);
76  Scalar sqrt_scale = sqrt(scale);
77  Eigen::Vector<Scalar, kNumParams> quat =
78  Rotation3Impl<Scalar>::exp(vec_omega);
79  quat *= sqrt_scale;
80  return quat;
81  }

127  static auto multiplication(
128  Eigen::Vector<Scalar, kNumParams> const& lhs_params,
129  Eigen::Vector<Scalar, kNumParams> const& rhs_params)
130  -> Eigen::Vector<Scalar, kNumParams> {
131  auto result =
132  QuaternionImpl<Scalar>::multiplication(lhs_params, rhs_params);
133  Scalar const squared_scale = result.squaredNorm();
134 
135  if (squared_scale < kEpsilon<Scalar> * kEpsilon<Scalar>) {
136  /// Saturation to ensure class invariant.
137  result.normalize();
138  result *= kEpsilonPlus<Scalar>;
139  }
140  if (squared_scale > Scalar(1.) / (kEpsilon<Scalar> * kEpsilon<Scalar>)) {
141  /// Saturation to ensure class invariant.
142  result.normalize();
143  result /= kEpsilonPlus<Scalar>;
144  }
145  return result;
146  }

172  static auto compactMatrix(
173  Eigen::Vector<Scalar, kNumParams> const& non_zero_quat)
174  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
175  Eigen::Matrix<Scalar, kPointDim, kPointDim> s_r;
176 
177  Scalar const vx_sq = non_zero_quat.x() * non_zero_quat.x();
178  Scalar const vy_sq = non_zero_quat.y() * non_zero_quat.y();
179  Scalar const vz_sq = non_zero_quat.z() * non_zero_quat.z();
180  Scalar const w_sq = non_zero_quat.w() * non_zero_quat.w();
181  Scalar const two_vx = Scalar(2) * non_zero_quat.x();
182  Scalar const two_vy = Scalar(2) * non_zero_quat.y();
183  Scalar const two_vz = Scalar(2) * non_zero_quat.z();
184  Scalar const two_vx_vy = two_vx * non_zero_quat.y();
185  Scalar const two_vx_vz = two_vx * non_zero_quat.z();
186  Scalar const two_vx_w = two_vx * non_zero_quat.w();
187  Scalar const two_vy_vz = two_vy * non_zero_quat.z();
188  Scalar const two_vy_w = two_vy * non_zero_quat.w();
189  Scalar const two_vz_w = two_vz * non_zero_quat.w();
190 
191  s_r(0, 0) = vx_sq - vy_sq - vz_sq + w_sq;
192  s_r(1, 0) = two_vx_vy + two_vz_w;
193  s_r(2, 0) = two_vx_vz - two_vy_w;
194 
195  s_r(0, 1) = two_vx_vy - two_vz_w;
196  s_r(1, 1) = -vx_sq + vy_sq - vz_sq + w_sq;
197  s_r(2, 1) = two_vx_w + two_vy_vz;
198 
199  s_r(0, 2) = two_vx_vz + two_vy_w;
200  s_r(1, 2) = -two_vx_w + two_vy_vz;
201  s_r(2, 2) = -vx_sq - vy_sq + vz_sq + w_sq;
202  return s_r;
203  }

248  static auto dxExpX(Eigen::Vector<Scalar, kDof> const& a)
249  -> Eigen::Matrix<Scalar, kNumParams, kDof> {
250  using std::exp;
251  using std::sqrt;
252  Eigen::Matrix<Scalar, 4, 4> j;
253  Eigen::Vector3<Scalar> const omega = a.template head<3>();
254  Scalar const sigma = a[3];
255  Eigen::Vector<Scalar, kNumParams> quat = Rotation3Impl<Scalar>::exp(omega);
256  Scalar const scale = sqrt(exp(sigma));
257  Scalar const scale_half = scale * Scalar(0.5);
258 
259  j.template block<4, 3>(0, 0) = Rotation3Impl<Scalar>::dxExpX(omega) * scale;
260  j.col(3) = a * scale_half;
261  return j;
262  }

277  Eigen::Vector<Scalar, kNumParams> const& non_zero_quat)
278  -> Eigen::Matrix<Scalar, kNumParams, kDof> {

281  Scalar const c0 = Scalar(0.5) * non_zero_quat.w();
282  Scalar const c1 = Scalar(0.5) * non_zero_quat.z();
283  Scalar const c2 = -c1;

285  Scalar const c4 = Scalar(0.5) * non_zero_quat.x();
286  Scalar const c5 = -c4;
287  Scalar const c6 = -c3;
288  j(0, 0) = c0;
289  j(0, 1) = c2;
290  j(0, 2) = c3;
291  j(1, 0) = c1;
292  j(1, 1) = c0;
293  j(1, 2) = c5;
294  j(2, 0) = c6;
295  j(2, 1) = c4;
296  j(2, 2) = c0;

303  static auto dxLogThisInvTimesXAtThis(
304  Eigen::Vector<Scalar, kNumParams> const& q)
305  -> Eigen::Matrix<Scalar, kDof, kNumParams> {
306  Eigen::Matrix<Scalar, 4, 4> j;
307  // clang-format off
308  j << q.w(), q.z(), -q.y(), -q.x(),
309  -q.z(), q.w(), q.x(), -q.y(),

313  const Scalar scaler = Scalar(2.) / q.squaredNorm();
314  return j * scaler;
