49  static auto multiplication(
50  Eigen::Vector<Scalar, 4> const& lhs, Eigen::Vector<Scalar, 4> const& rhs)
51  -> Eigen::Vector<Scalar, 4> {
52  Eigen::Vector3<Scalar> lhs_ivec = lhs.template head<3>();
53  Eigen::Vector3<Scalar> rhs_ivec = rhs.template head<3>();
54 
55  Eigen::Vector<Scalar, 4> out;

57  out.template head<3>() =
58  lhs.w() * rhs_ivec + rhs.w() * lhs_ivec + lhs_ivec.cross(rhs_ivec);
