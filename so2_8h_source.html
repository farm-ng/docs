61 /// So2 base type - implements So2 class but is storage agnostic.
62 ///
63 /// SO(2) is the group of rotations in 2d. As a matrix group, it is the set of
64 /// matrices which are orthogonal such that ``R * R' = I`` (with ``R'`` being
65 /// the transpose of ``R``) and have a positive determinant. In particular, the
66 /// determinant is 1. Let ``theta`` be the rotation angle, the rotation matrix
67 /// can be written in close form:
68 ///
69 /// | cos(theta) -sin(theta) |
70 /// | sin(theta) cos(theta) |
71 ///
72 /// As a matter of fact, the first column of those matrices is isomorph to the
73 /// set of unit complex numbers U(1). Thus, internally, So2 is represented as
74 /// complex number with length 1.
75 ///
76 /// SO(2) is a compact and commutative group. First it is compact since the set
77 /// of rotation matrices is a closed and bounded set. Second it is commutative
78 /// since ``R(alpha) * R(beta) = R(beta) * R(alpha)``, simply because ``alpha +
79 /// beta = beta + alpha`` with ``alpha`` and ``beta`` being rotation angles
80 /// (about the same axis).
81 ///
82 /// Class invariant: The 2-norm of ``unit_complex`` must be close to 1.
83 /// Technically speaking, it must hold that:
84 ///
85 /// ``|unit_complex().squaredNorm() - 1| <= Constants::epsilon()``.
86 template <class TDerived>
87 class So2Base {
88  public:
89  using Scalar = typename Eigen::internal::traits<TDerived>::Scalar;
90  using ComplexT = typename Eigen::internal::traits<TDerived>::ComplexType;
91  using ComplexTemporaryType = Eigen::Matrix<Scalar, 2, 1>;

94  /// since we only have in-plane rotations).

97  static int constexpr kNumParameters = 2;

127  /// Adjoint transformation
128  ///
129  /// This function return the adjoint transformation ``Ad`` of the group
130  /// element ``A`` such that for all ``x`` it holds that
131  /// ``hat(Ad_A * x) = A * hat(x) A^{-1}``. See hat-operator below.
132  ///
133  /// It simply ``1``, since ``SO(2)`` is a commutative group.
134  ///
135  SOPHUS_FUNC [[nodiscard]] Adjoint adj() const { return Scalar(1); }

163  /// Logarithmic map
164  ///
165  /// Computes the logarithm, the inverse of the group exponential which maps
166  /// element of the group (rotation matrices) to elements of the tangent space
167  /// (rotation angles).
168  ///
169  /// To be specific, this function computes ``vee(logmat(.))`` with
170  /// ``logmat(.)`` being the matrix logarithm and ``vee(.)`` the vee-operator
171  /// of SO(2).
172  ///
173  SOPHUS_FUNC [[nodiscard]] Scalar log() const {
174  using std::atan2;
175  return atan2(unitComplex().y(), unitComplex().x());
176  }

178  /// It re-normalizes ``unit_complex`` to unit length.
179  ///
180  /// Note: Because of the class invariant, there is typically no need to call
181  /// this function directly.
182  ///
183  SOPHUS_FUNC void normalize() {
184  using std::hypot;
185  // Avoid under/overflows for higher precision
186  Scalar length = hypot(unitComplex().x(), unitComplex().y());
187  SOPHUS_ASSERT(
188  length >= kEpsilon<Scalar>,
189  "Complex number should not be close to zero!");
190  mutUnitComplex() /= length;
191  }
192 
193  /// Returns 2x2 matrix representation of the instance.
194  ///
195  /// For SO(2), the matrix representation is an orthogonal matrix ``R`` with
196  /// ``det(R)=1``, thus the so-called "rotation matrix".
197  ///
198  SOPHUS_FUNC [[nodiscard]] Transformation matrix() const {
199  Scalar const& real = unitComplex().x();
200  Scalar const& imag = unitComplex().y();
201  Transformation mat_r;
202  // clang-format off
203  mat_r <<
204  real, -imag,
205  imag, real;
206  // clang-format on
207  return mat_r;
208  }

219  /// Group multiplication, which is rotation concatenation.
220  ///
221  template <class TOtherDerived>
222  SOPHUS_FUNC So2Product<TOtherDerived> operator*(
223  So2Base<TOtherDerived> const& other) const {
224  using ResultT = ReturnScalar<TOtherDerived>;
225  Scalar const lhs_real = unitComplex().x();
226  Scalar const lhs_imag = unitComplex().y();
227  typename TOtherDerived::Scalar const& rhs_real = other.unitComplex().x();
228  typename TOtherDerived::Scalar const& rhs_imag = other.unitComplex().y();
229  // complex multiplication
230  ResultT const result_real = lhs_real * rhs_real - lhs_imag * rhs_imag;
231  ResultT const result_imag = lhs_real * rhs_imag + lhs_imag * rhs_real;
232 
233  ResultT const squared_norm =
234  result_real * result_real + result_imag * result_imag;
235  // We can assume that the squared-norm is close to 1 since we deal with a
236  // unit complex number. Due to numerical precision issues, there might
237  // be a small drift after pose concatenation. Hence, we need to renormalizes
238  // the complex number here.
239  // Since squared-norm is close to 1, we do not need to calculate the costly
240  // square-root, but can use an approximation around 1 (see
241  // http://stackoverflow.com/a/12934750 for details).
242  if (squared_norm != ResultT(1.0)) {
243  ResultT const scale = ResultT(2.0) / (ResultT(1.0) + squared_norm);
244  return So2Product<TOtherDerived>(
245  result_real * scale, result_imag * scale);
246  }
247  return So2Product<TOtherDerived>(result_real, result_imag);
248  }

250  /// Group action on 2-points.
251  ///
252  /// This function rotates a 2 dimensional point ``p`` by the So2 element
253  /// ``bar_R_foo`` (= rotation matrix): ``p_bar = bar_R_foo * p_foo``.
254  ///
255  template <
256  typename TPointDerived,
257  typename = typename std::enable_if<
258  IsFixedSizeVector<TPointDerived, 2>::value>::type>
259  SOPHUS_FUNC PointProduct<TPointDerived> operator*(
260  Eigen::MatrixBase<TPointDerived> const& p) const {
261  Scalar const& real = unitComplex().x();
262  Scalar const& imag = unitComplex().y();
263  return PointProduct<TPointDerived>(
264  real * p[0] - imag * p[1], //
265  imag * p[0] + real * p[1]);
266  }
267 
268  /// Group action on homogeneous 2-points.
269  ///
270  /// This function rotates a homogeneous 2 dimensional point ``p`` by the So2
271  /// element ``bar_R_foo`` (= rotation matrix): ``p_bar = bar_R_foo * p_foo``.
272  ///
273  template <
274  typename THPointDerived,
275  typename = typename std::enable_if<
276  IsFixedSizeVector<THPointDerived, 3>::value>::type>
277  SOPHUS_FUNC HomogeneousPointProduct<THPointDerived> operator*(
278  Eigen::MatrixBase<THPointDerived> const& p) const {
279  Scalar const& real = unitComplex().x();
280  Scalar const& imag = unitComplex().y();
281  return HomogeneousPointProduct<THPointDerived>(
282  real * p[0] - imag * p[1], //
283  imag * p[0] + real * p[1], //
284  p[2]);
285  }

287  /// Group action on lines.
288  ///
289  /// This function rotates a parametrized line ``l(t) = o + t * d`` by the So2
290  /// element:
291  ///
292  /// Both direction ``d`` and origin ``o`` are rotated as a 2 dimensional point
293  ///
294  SOPHUS_FUNC Line operator*(Line const& l) const {
295  return Line((*this) * l.origin(), (*this) * l.direction());
296  }
297 
298  /// Group action on hyper-planes.
299  ///
300  /// This function rotates a hyper-plane ``n.x + d = 0`` by the So2
301  /// element:
302  ///
303  /// Normal vector ``n`` is rotated
304  /// Offset ``d`` is left unchanged
305  ///
306  /// Note that in 2d-case hyper-planes are just another parametrization of
307  /// lines
308  ///
309  SOPHUS_FUNC Hyperplane operator*(Hyperplane const& p) const {
310  return Hyperplane((*this) * p.normal(), p.offset());
311  }

418  /// Constructor from rotation matrix
419  ///
420  /// Precondition: rotation matrix need to be orthogonal with determinant of 1.

467  /// Group exponential
468  ///
469  /// This functions takes in an element of tangent space (= rotation angle
470  /// ``theta``) and returns the corresponding element of the group SO(2).
471  ///
472  /// To be more specific, this function computes ``expmat(hat(omega))``
473  /// with ``expmat(.)`` being the matrix exponential and ``hat(.)`` being the
474  /// hat()-operator of SO(2).
475  ///
476  SOPHUS_FUNC static So2<Scalar> exp(Tangent const& theta) {
477  using std::cos;
478  using std::sin;
479  return So2<Scalar>(cos(theta), sin(theta));
480  }

482  /// Returns derivative of exp(x) wrt. x.
483  ///
484  SOPHUS_FUNC static Eigen::Matrix<Scalar, kNumParameters, kDoF> dxExpX(
485  Tangent const& theta) {

487  using std::sin;
488  return Eigen::Matrix<Scalar, kNumParameters, kDoF>(-sin(theta), cos(theta));
489  }

492  ///

494  return Eigen::Matrix<Scalar, kNumParameters, kDoF>(Scalar(0), Scalar(1));

504  /// Returns derivative of ``expmat(x)`` wrt. ``x_i at x=0``, with

506  ///

509  }

511  /// Returns the infinitesimal generators of SO(2).
512  ///
513  /// The infinitesimal generators of SO(2) is:
514  ///
515  /// | 0 -1 |
516  /// | 1 0 |
517  ///
518  SOPHUS_FUNC static Transformation generator() { return hat(Scalar(1)); }
519 
520  /// hat-operator
521  ///
522  /// It takes in the scalar representation ``theta`` (= rotation angle) and
523  /// returns the corresponding matrix representation of Lie algebra element.
524  ///
525  /// Formally, the hat()-operator of SO(2) is defined as
526  ///
527  /// ``hat(.): R^2 -> R^{2x2}, hat(theta) = theta * G``
528  ///
529  /// with ``G`` being the infinitesimal generator of SO(2).
530  ///
531  /// The corresponding inverse is the vee()-operator, see below.
532  ///
533  SOPHUS_FUNC static Transformation hat(Tangent const& theta) {
534  Transformation omega;
535  // clang-format off
536  omega <<
537  Scalar(0), -theta,
538  theta, Scalar(0);
539  // clang-format on
540  return omega;
541  }

543  /// Returns closed So2 given arbitrary 2x2 matrix.
544  ///
545  template <class TS = Scalar>
546  static SOPHUS_FUNC std::enable_if_t<std::is_floating_point<TS>::value, So2>
547  fitToSo2(Transformation const& r) {
548  return So2(makeRotationMatrix(r));
549  }

551  /// Lie bracket
552  ///
553  /// It returns the Lie bracket of SO(2). Since SO(2) is a commutative group,
554  /// the Lie bracket is simple ``0``.
555  ///
556  SOPHUS_FUNC static Tangent lieBracket(
557  Tangent const& /*unused*/, Tangent const& /*unused*/) {
558  return Scalar(0);
559  }

561  /// Draw uniform sample from SO(2) manifold.
562  ///
563  template <class TUniformRandomBitGenerator>
564  static So2 sampleUniform(TUniformRandomBitGenerator& generator) {
565  static_assert(
566  kIsUniformRandomBitGeneratorV<TUniformRandomBitGenerator>,
567  "generator must meet the UniformRandomBitGenerator concept");
568  std::uniform_real_distribution<Scalar> uniform(-kPi<Scalar>, kPi<Scalar>);
569  return So2(uniform(generator));
570  }
571 
572  /// vee-operator
573  ///
574  /// It takes the 2x2-matrix representation ``Omega`` and maps it to the
575  /// corresponding scalar representation of Lie algebra.
576  ///
577  /// This is the inverse of the hat()-operator, see above.
578  ///
579  /// Precondition: ``Omega`` must have the following structure:
580  ///
581  /// | 0 -a |
582  /// | a 0 |
583  ///
584  SOPHUS_FUNC static Tangent vee(Transformation const& omega) {
585  using std::abs;
586  return omega(1, 0);
587  }

594  ComplexMember unit_complex_; // NOLINT
