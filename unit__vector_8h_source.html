46  static Expected<UnitVector> tryFromUnitVector(
47  Eigen::Matrix<TScalar, kN, 1> const& v) {
48  using std::abs;
49  SOPHUS_ASSERT_OR_ERROR(
50  abs(v.squaredNorm() - TScalar(1.0)) < sophus::kEpsilon<TScalar>,
51  "v must be unit length is {}.\n{}",
52  v.squaredNorm(),
53  v.transpose());
54 
55  UnitVector unit_vector;
56  unit_vector.vector_ = v;
57  return unit_vector;
58  }

60  static UnitVector fromVectorAndNormalize(

62  return fromUnitVector(v.normalized());

65  [[nodiscard]] Eigen::Matrix<TScalar, kN, 1> const& vector() const {

110 /// Construct rotation which would take unit direction vector ``from`` into
111 /// ``to`` such that ``to = rotThroughPoints(from,to) * from``. I.e. that the
112 /// rotated point ``from`` is colinear with ``to`` (equal up to scale)
113 ///
114 /// The axis of rotation is perpendicular to both ``from`` and ``to``.
115 ///
116 template <class TScalar>
117 SOPHUS_FUNC So3<TScalar> rotThroughPoints(
118  UnitVector3<TScalar> const& from, UnitVector3<TScalar> const& to) {
119  using std::abs;
120  using std::atan2;
121  Eigen::Vector<TScalar, 3> from_cross_to = from.vector().cross(to.vector());
122  TScalar n = from_cross_to.norm();
123  if (abs(n) < sophus::kEpsilon<TScalar>) {
124  return So3<TScalar>();
125  }
126  // https://stackoverflow.com/a/32724066
127  TScalar angle = atan2(n, from.vector().dot(to.vector()));
128 
129  return So3<TScalar>::exp(angle * from_cross_to / n);
130 }

132 /// Construct rotation which would take direction vector ``from`` into ``to``
133 /// such that ``to \propto rotThroughPoints(from,to) * from``. I.e. that the
134 /// rotated point ``from`` is colinear with ``to`` (equal up to scale)
135 ///
136 /// The axis of rotation is perpendicular to both ``from`` and ``to``.
137 ///
138 /// Precondition: Neither ``from`` nor ``to`` must be zero. This is unchecked.
139 template <class TScalar>
140 SOPHUS_FUNC So3<TScalar> rotThroughPoints(
141  Eigen::Vector<TScalar, 3> const& from,
142  Eigen::Vector<TScalar, 3> const& to) {
143  using std::abs;
144  using std::atan2;
145  Eigen::Vector<TScalar, 3> from_cross_to = from.cross(to);
146  TScalar n = from_cross_to.norm();
147  if (abs(n) < sophus::kEpsilon<TScalar>) {
148  return So3<TScalar>();
149  }
150  // https://stackoverflow.com/a/32724066
151  TScalar angle = atan2(n, from.dot(to));
152 
153  return So3<TScalar>::exp(angle * from_cross_to / n);
154 }
