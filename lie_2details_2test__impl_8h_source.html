100  // For the time being, leftJacobian and leftJacobianInverse are only
101  // implemented for So3 and Se3
102  template <class TG = LieGroup>
103  std::enable_if_t<
104  std::is_same<TG, So3<Scalar>>::value ||
105  std::is_same<TG, Se3<Scalar>>::value,
106  bool>
107  leftJacobianTest() {
108  bool passed = true;
109  for (auto const& x : tangent_vec_) {
110  LieGroup const inv_exp_x = LieGroup::exp(x).inverse();
111 
112  // Explicit implement the derivative in the Lie Group in first principles
113  // as a vector field: D_x f(x) = D_h log(f(x + h) . f(x)^{-1})
114  Eigen::Matrix<Scalar, kDoF, kDoF> const j_num =
115  vectorFieldNumDiff<Scalar, kDoF, kDoF>(
116  [&inv_exp_x](Tangent const& x_plus_delta) {
117  return (LieGroup::exp(x_plus_delta) * inv_exp_x).log();
118  },
119  x);
120 
121  // Analytical left Jacobian
122  Eigen::Matrix<Scalar, kDoF, kDoF> const j = LieGroup::leftJacobian(x);
123  SOPHUS_TEST_APPROX(
124  passed, j, j_num, Scalar(100) * small_eps_sqrt, "Left Jacobian");
125 
126  Eigen::Matrix<Scalar, kDoF, kDoF> j_inv =
127  LieGroup::leftJacobianInverse(x);
128 
129  SOPHUS_TEST_APPROX(
130  passed,
131  j,
132  j_inv.inverse().eval(),
133  Scalar(100) * small_eps_sqrt,
134  "Left Jacobian and its analytical Inverse");
135  }

137  return passed;
