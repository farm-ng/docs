56 /// Sim2 base type - implements Sim2 class but is storage agnostic.
57 ///
58 /// Sim(2) is the group of rotations and translation and scaling in 2d. It is
59 /// the semi-direct product of R+xSO(2) and the 2d Euclidean vector space. The
60 /// class is represented using a composition of RxSo2 for scaling plus
61 /// rotation and a 2-vector for translation.
62 ///
63 /// - 3x3 Eigen::Matrix representation:
64 ///
65 /// ```
66 /// | s*R t |
67 /// | o 1 |
68 /// ```
69 ///
70 /// where ``R`` is a 3x3 rotation matrix, ``s`` a positive scale factor,
71 /// ``t`` a translation 3-vector and ``o`` a 3-column vector of zeros.
72 ///
73 /// - Tangent 4-vector: [upsilon, omega, sigma],
74 ///
75 /// where ``upsilon`` is the translational velocity 3-vector and ``omega`` the
76 /// rotational velocity 3-vector, and ``sigma = log(s)``.
77 ///
78 /// - Internal 4-vector representation: [t0, t1, zr, zi],
79 ///
80 /// with ``t0, t1`` are the translational components, and ``zr``
81 /// the real part and ``zi`` the imaginary part of a non-zero
82 /// complex number. Here the scale ``s`` is equal to the norm of the complex
83 /// number ``s = |z|``.
84 ///
85 /// See RxSo2 for more details of the scaling + rotation representation in 2d.
86 ///
87 ///
88 /// Sim(2) is neither compact, nor a commutative group.
89 ///
90 /// See RxSo2 for more details of the scaling + rotation representation in 2d.
91 ///
92 template <class TDerived>
93 class Sim2Base {
94  public:
95  using Scalar = typename Eigen::internal::traits<TDerived>::Scalar;
96  using TranslationType =
97  typename Eigen::internal::traits<TDerived>::TranslationType;
98  using RxSo2Type = typename Eigen::internal::traits<TDerived>::RxSo2Type;

135  /// Adjoint transformation
136  ///
137  /// This function return the adjoint transformation ``Ad`` of the group
138  /// element ``A`` such that for all ``x`` it holds that
139  /// ``hat(Ad_A * x) = A * hat(x) A^{-1}``. See hat-operator below.
140  ///
141  SOPHUS_FUNC [[nodiscard]] Adjoint adj() const {
142  Adjoint res;
143  res.setZero();
144  res.template block<2, 2>(0, 0) = rxso2().matrix();
145  res(0, 2) = translation()[1];
146  res(1, 2) = -translation()[0];
147  res.template block<2, 1>(0, 3) = -translation();
148 
149  res(2, 2) = Scalar(1);
150 
151  res(3, 3) = Scalar(1);
152  return res;
153  }

164  /// Returns group inverse.
165  ///
166  SOPHUS_FUNC [[nodiscard]] Sim2<Scalar> inverse() const {
167  RxSo2<Scalar> inv_r = rxso2().inverse();

171  /// Logarithmic map
172  ///
173  /// Computes the logarithm, the inverse of the group exponential which maps
174  /// element of the group (rigid body transformations) to elements of the
175  /// tangent space (twist).
176  ///
177  /// To be specific, this function computes ``vee(logmat(.))`` with
178  /// ``logmat(.)`` being the matrix logarithm and ``vee(.)`` the vee-operator
179  /// of Sim(2).
180  ///
181  SOPHUS_FUNC [[nodiscard]] Tangent log() const {
182  /// The derivation of the closed-form Sim(2) logarithm for is done
183  /// analogously to the closed-form solution of the SE(2) logarithm, see
184  /// J. Gallier, D. Xu, "Computing exponentials of skew symmetric matrices
185  /// and logarithms of orthogonal matrices", IJRA 2002.
186  /// https:///pdfs.semanticscholar.org/cfe3/e4b39de63c8cabd89bf3feff7f5449fc981d.pdf
187  /// (Sec. 6., pp. 8)
188  Tangent res;
189  Eigen::Vector2<Scalar> const theta_sigma = rxso2().log();
190  Scalar const theta = theta_sigma[0];
191  Scalar const sigma = theta_sigma[1];
192  Eigen::Matrix2<Scalar> const omega = So2<Scalar>::hat(theta);
193  Eigen::Matrix2<Scalar> const w_inv =
194  details::calcWInv<Scalar, 2>(omega, theta, sigma, scale());
195 
196  res.segment(0, 2) = w_inv * translation();
197  res[2] = theta;
198  res[3] = sigma;
199  return res;
200  }

202  /// Returns 3x3 matrix representation of the instance.
203  ///
204  /// It has the following form:
205  ///
206  /// | s*R t |
207  /// | o 1 |
208  ///
209  /// where ``R`` is a 2x2 rotation matrix, ``s`` a scale factor, ``t`` a
210  /// translation 2-vector and ``o`` a 2-column vector of zeros.
211  ///
212  SOPHUS_FUNC [[nodiscard]] Transformation matrix() const {
213  Transformation homogenious_matrix;
214  homogenious_matrix.template topLeftCorner<2, 3>() = matrix2x3();
215  homogenious_matrix.row(2) =
216  Eigen::Matrix<Scalar, 3, 1>(Scalar(0), Scalar(0), Scalar(1));
217  return homogenious_matrix;
218  }

224  matrix.template topLeftCorner<2, 2>() = rxso2().matrix();
225  matrix.col(2) = translation();
226  return matrix;

251  /// Group action on 2-points.
252  ///
253  /// This function rotates, scales and translates a two dimensional point
254  /// ``p`` by the Sim(2) element ``(bar_sR_foo, t_bar)`` (= similarity
255  /// transformation):
256  ///
257  /// ``p_bar = bar_sR_foo * p_foo + t_bar``.
258  ///
259  template <
260  typename TPointDerived,
261  typename = typename std::enable_if<
262  IsFixedSizeVector<TPointDerived, 2>::value>::type>
263  SOPHUS_FUNC PointProduct<TPointDerived> operator*(
264  Eigen::MatrixBase<TPointDerived> const& p) const {
265  return rxso2() * p + translation();
266  }
267 
268  /// Group action on homogeneous 2-points. See above for more details.
269  ///
270  template <
271  typename THPointDerived,
272  typename = typename std::enable_if<
273  IsFixedSizeVector<THPointDerived, 3>::value>::type>
274  SOPHUS_FUNC HomogeneousPointProduct<THPointDerived> operator*(
275  Eigen::MatrixBase<THPointDerived> const& p) const {
276  PointProduct<THPointDerived> const tp =
277  rxso2() * p.template head<2>() + p(2) * translation();
278  return HomogeneousPointProduct<THPointDerived>(tp(0), tp(1), p(2));
279  }

281  /// Group action on lines.
282  ///
283  /// This function rotates, scales and translates a parametrized line
284  /// ``l(t) = o + t * d`` by the Sim(2) element:
285  ///
286  /// Origin ``o`` is rotated, scaled and translated
287  /// Direction ``d`` is rotated
288  ///
289  SOPHUS_FUNC Line operator*(Line const& l) const {
290  Line rotated_line = rxso2() * l;
291  return Line(
292  rotated_line.origin() + translation(), rotated_line.direction());
293  }
294 
295  /// Group action on hyper-planes.
296  ///
297  /// This function rotates a hyper-plane ``n.x + d = 0`` by the Sim2
298  /// element:
299  ///
300  /// Normal vector ``n`` is rotated
301  /// Offset ``d`` is scaled and adjusted for translation
302  ///
303  /// Note that in 2d-case hyper-planes are just another parametrization of
304  /// lines
305  ///
306  SOPHUS_FUNC Hyperplane operator*(Hyperplane const& p) const {
307  Hyperplane const rotated = rxso2() * p;
308  return Hyperplane(
309  rotated.normal(),
310  rotated.offset() - translation().dot(rotated.normal()));
311  }

338  /// Returns derivative of this * Sim2::exp(x) wrt. x at x=0.
339  ///
340  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kNumParameters, kDoF>

342  Eigen::Matrix<Scalar, kNumParameters, kDoF> d;
343  d.template block<2, 2>(0, 0).setZero();

351  ///
352  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kDoF, kNumParameters>

354  Eigen::Matrix<Scalar, kNumParameters, kDoF> d;

553  /// Returns derivative of exp(x) wrt. x_i at x=0.
554  ///
555  SOPHUS_FUNC static Eigen::Matrix<Scalar, kNumParameters, kDoF> dxExpXAt0() {
556  Eigen::Matrix<Scalar, kNumParameters, kDoF> j;
557  j.template block<2, 2>(0, 0).setZero();
558  j.template block<2, 2>(0, 2) = RxSo2<Scalar>::dxExpXAt0();
559  j.template block<2, 2>(2, 0).setIdentity();
560  j.template block<2, 2>(2, 2).setZero();

564  /// Returns derivative of exp(x) wrt. x.
565  ///
566  SOPHUS_FUNC static Eigen::Matrix<Scalar, kNumParameters, kDoF> dxExpX(
567  Tangent const& vec_a) {
568  static Eigen::Matrix2<Scalar> const kI = Eigen::Matrix2<Scalar>::Identity();
569  static Scalar const kOne(1.0);
570 
571  Scalar const theta = vec_a[2];
572  Scalar const sigma = vec_a[3];
573 
574  Eigen::Matrix2<Scalar> const omega = So2<Scalar>::hat(theta);
575  Eigen::Matrix2<Scalar> const omega_dtheta = So2<Scalar>::hat(kOne);
576  Eigen::Matrix2<Scalar> const omega2 = omega * omega;
577  Eigen::Matrix2<Scalar> const omega2_dtheta =
578  omega_dtheta * omega + omega * omega_dtheta;
579  Eigen::Matrix2<Scalar> const w =
580  details::calcW<Scalar, 2>(omega, theta, sigma);
581  Eigen::Vector2<Scalar> const upsilon = vec_a.segment(0, 2);
582 
583  Eigen::Matrix<Scalar, kNumParameters, kDoF> j;
584  j.template block<2, 2>(0, 0).setZero();
585  j.template block<2, 2>(0, 2) =
586  RxSo2<Scalar>::dxExpX(vec_a.template tail<2>());
587  j.template block<2, 2>(2, 0) = w;
588 
589  Scalar a;
590  Scalar b;
591  Scalar c;
592  Scalar a_dtheta;
593  Scalar b_dtheta;
594  Scalar a_dsigma;
595  Scalar b_dsigma;
596  Scalar c_dsigma;
597  details::calcWDerivatives(
598  theta,
599  sigma,
600  a,
601  b,
602  c,
603  a_dsigma,
604  b_dsigma,
605  c_dsigma,
606  a_dtheta,
607  b_dtheta);
608 
609  j.template block<2, 1>(2, 2) = (a_dtheta * omega + a * omega_dtheta +
610  b_dtheta * omega2 + b * omega2_dtheta) *
611  upsilon;
612  j.template block<2, 1>(2, 3) =
613  (a_dsigma * omega + b_dsigma * omega2 + c_dsigma * kI) * upsilon;
614 
615  return j;
616  }

630  ///
631  SOPHUS_FUNC static Transformation dxiExpmatXAt0(int i) {
632  return generator(i);
633  }

635  /// Derivative of Lie bracket with respect to first element.
636  ///
637  /// This function returns ``D_a [a, b]`` with ``D_a`` being the
638  /// differential operator with respect to ``a``, ``[a, b]`` being the lie
639  /// bracket of the Lie algebra sim(2).
640  /// See ``lieBracket()`` below.
641  ///
642 
643  /// Group exponential
644  ///
645  /// This functions takes in an element of tangent space and returns the
646  /// corresponding element of the group Sim(2).
647  ///
648  /// The first two components of ``a`` represent the translational part
649  /// ``upsilon`` in the tangent space of Sim(2), the following two components
650  /// of ``a`` represents the rotation ``theta`` and the final component
651  /// represents the logarithm of the scaling factor ``sigma``.
652  /// To be more specific, this function computes ``expmat(hat(a))`` with
653  /// ``expmat(.)`` being the matrix exponential and ``hat(.)`` the hat-operator
654  /// of Sim(2), see below.
655  ///
656  SOPHUS_FUNC static Sim2<Scalar> exp(Tangent const& vec_a) {
657  // For the derivation of the exponential map of Sim(kMatrixDim) see
658  // H. Strasdat, "Local Accuracy and Global Consistency for Efficient Visual
659  // SLAM", PhD thesis, 2012.
660  // http:///hauke.strasdat.net/files/strasdat_thesis_2012.pdf (A.5, pp. 186)
661  Eigen::Vector2<Scalar> const upsilon = vec_a.segment(0, 2);
662  Scalar const theta = vec_a[2];
663  Scalar const sigma = vec_a[3];
664  RxSo2<Scalar> rxso2 = RxSo2<Scalar>::exp(vec_a.template tail<2>());
665  Eigen::Matrix2<Scalar> const omega = So2<Scalar>::hat(theta);
666  Eigen::Matrix2<Scalar> const w =
667  details::calcW<Scalar, 2>(omega, theta, sigma);
668  return Sim2<Scalar>(rxso2, w * upsilon);
669  }
670 
671  /// Returns the ith infinitesimal generators of Sim(2).
672  ///
673  /// The infinitesimal generators of Sim(2) are:
674  ///
675  /// ```
676  /// | 0 0 1 |
677  /// G_0 = | 0 0 0 |
678  /// | 0 0 0 |
679  ///
680  /// | 0 0 0 |
681  /// G_1 = | 0 0 1 |
682  /// | 0 0 0 |
683  ///
684  /// | 0 -1 0 |
685  /// G_2 = | 1 0 0 |
686  /// | 0 0 0 |
687  ///
688  /// | 1 0 0 |
689  /// G_3 = | 0 1 0 |
690  /// | 0 0 0 |
691  /// ```
692  ///
693  /// Precondition: ``i`` must be in [0, 3].
694  ///
695  SOPHUS_FUNC static Transformation generator(int i) {
696  SOPHUS_ASSERT(i >= 0 || i <= 3, "i should be in range [0,3].");
697  Tangent e;
698  e.setZero();
699  e[i] = Scalar(1);
700  return hat(e);
701  }
702 
703  /// hat-operator
704  ///
705  /// It takes in the 4-vector representation and returns the corresponding
706  /// matrix representation of Lie algebra element.
707  ///
708  /// Formally, the hat()-operator of Sim(2) is defined as
709  ///
710  /// ``hat(.): R^4 -> R^{3x3}, hat(a) = sum_i a_i * G_i`` (for i=0,...,6)
711  ///
712  /// with ``G_i`` being the ith infinitesimal generator of Sim(2).
713  ///
714  /// The corresponding inverse is the vee()-operator, see below.
715  ///
716  SOPHUS_FUNC static Transformation hat(Tangent const& vec_a) {
717  Transformation omega;
718  omega.template topLeftCorner<2, 2>() =
719  RxSo2<Scalar>::hat(vec_a.template tail<2>());
720  omega.col(2).template head<2>() = vec_a.template head<2>();
721  omega.row(2).setZero();
722  return omega;
723  }
724 
725  /// Lie bracket
726  ///
727  /// It computes the Lie bracket of Sim(2). To be more specific, it computes
728  ///
729  /// ``[omega_1, omega_2]_sim2 := vee([hat(omega_1), hat(omega_2)])``
730  ///
731  /// with ``[A,B] := AB-BA`` being the matrix commutator, ``hat(.)`` the
732  /// hat()-operator and ``vee(.)`` the vee()-operator of Sim(2).
733  ///
734  SOPHUS_FUNC static Tangent lieBracket(
735  Tangent const& vec_a, Tangent const& vec_b) {
736  Eigen::Vector2<Scalar> const vec_upsilon1 = vec_a.template head<2>();
737  Eigen::Vector2<Scalar> const vec_upsilon2 = vec_b.template head<2>();
738  Scalar const theta1 = vec_a[2];
739  Scalar const theta2 = vec_b[2];
740  Scalar const sigma1 = vec_a[3];
741  Scalar const sigma2 = vec_b[3];
742 
743  Tangent res;
744  res[0] = -theta1 * vec_upsilon2[1] + theta2 * vec_upsilon1[1] +
745  sigma1 * vec_upsilon2[0] - sigma2 * vec_upsilon1[0];
746  res[1] = theta1 * vec_upsilon2[0] - theta2 * vec_upsilon1[0] +
747  sigma1 * vec_upsilon2[1] - sigma2 * vec_upsilon1[1];
748  res[2] = Scalar(0);
749  res[3] = Scalar(0);
750 
751  return res;
752  }
753 
754  /// Draw uniform sample from Sim(2) manifold.
755  ///
756  /// Translations are drawn component-wise from the range [-1, 1].
757  /// The scale factor is drawn uniformly in log2-space from [-1, 1],
758  /// hence the scale is in [0.5, 2].
759  ///
760  template <class TUniformRandomBitGenerator>
761  static Sim2 sampleUniform(TUniformRandomBitGenerator& generator) {
762  std::uniform_real_distribution<Scalar> uniform(Scalar(-1), Scalar(1));
763  return Sim2(
764  RxSo2<Scalar>::sampleUniform(generator),
765  Eigen::Vector2<Scalar>(uniform(generator), uniform(generator)));
766  }
767 
768  /// vee-operator
769  ///
770  /// It takes the 3x3-matrix representation ``Omega`` and maps it to the
771  /// corresponding 4-vector representation of Lie algebra.
772  ///
773  /// This is the inverse of the hat()-operator, see above.
774  ///
775  /// Precondition: ``Omega`` must have the following structure:
776  ///
777  /// | d -c a |
778  /// | c d b |
779  /// | 0 0 0 |
780  ///
781  SOPHUS_FUNC static Tangent vee(Transformation const& mat_omega) {
782  Tangent upsilon_omega_sigma;
783  upsilon_omega_sigma.template head<2>() =
784  mat_omega.col(2).template head<2>();
785  upsilon_omega_sigma.template tail<2>() =
786  RxSo2<Scalar>::vee(mat_omega.template topLeftCorner<2, 2>());
787  return upsilon_omega_sigma;
788  }
789 
790  protected:
791  RxSo2Member rxso2_; // NOLINT
792  TranslationMember translation_; // NOLINT
793 };
