44  static auto areParamsValid(
45  Eigen::Vector<Scalar, kNumParams> const& unit_complex)
46  -> sophus::Expected<Success> {
47  static const Scalar kThr = kEpsilon<Scalar>;
48  const Scalar squared_norm = unit_complex.squaredNorm();
49  using std::abs;
50  if (!(abs(squared_norm - 1.0) <= kThr)) {

52  "complex number ({}, {}) is not unit length.\n"
53  "Squared norm: {}, thr: {}",
54  unit_complex[0],
55  unit_complex[1],

57  kThr);
58  }

100  static auto multiplication(
101  Eigen::Vector<Scalar, kNumParams> const& lhs_params,
102  Eigen::Vector<Scalar, kNumParams> const& rhs_params)
103  -> Eigen::Vector<Scalar, kNumParams> {
104  auto result = Complex::multiplication(lhs_params, rhs_params);
105  Scalar const squared_norm = result.squaredNorm();
106 
107  // We can assume that the squared-norm is close to 1 since we deal with a
108  // unit complex number. Due to numerical precision issues, there might
109  // be a small drift after pose concatenation. Hence, we need to renormalizes
110  // the complex number here.
111  // Since squared-norm is close to 1, we do not need to calculate the costly
112  // square-root, but can use an approximation around 1 (see
113  // http://stackoverflow.com/a/12934750 for details).
114  if (squared_norm != 1.0) {
115  Scalar const scale = 2.0 / (1.0 + squared_norm);
116  return scale * result;
117  }
118  return result;
119  }

157  // Sub-group concepts
158  static auto matV(
159  Eigen::Vector<Scalar, kNumParams> const& params,
160  Eigen::Vector<Scalar, kDof> const& theta)
161  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
162  Scalar sin_theta_by_theta;
163  Scalar one_minus_cos_theta_by_theta;
164  using std::abs;
165 
166  if (abs(theta[0]) < kEpsilon<Scalar>) {
167  Scalar theta_sq = theta[0] * theta[0];
168  sin_theta_by_theta = Scalar(1.) - Scalar(1. / 6.) * theta_sq;
169  one_minus_cos_theta_by_theta =
170  Scalar(0.5) * theta[0] - Scalar(1. / 24.) * theta[0] * theta_sq;
171  } else {
172  sin_theta_by_theta = params.y() / theta[0];
173  one_minus_cos_theta_by_theta = (Scalar(1.) - params.x()) / theta[0];
174  }
175  return Eigen::Matrix<Scalar, 2, 2>(
176  {{sin_theta_by_theta, -one_minus_cos_theta_by_theta},
177  {one_minus_cos_theta_by_theta, sin_theta_by_theta}});
178  }

180  static auto matVInverse(
181  Eigen::Vector<Scalar, kNumParams> const& z,
182  Eigen::Vector<Scalar, kDof> const& theta)
183  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
184  Scalar halftheta = Scalar(0.5) * theta[0];
185  Scalar halftheta_by_tan_of_halftheta;
186 
187  Scalar real_minus_one = z.x() - Scalar(1.);
188  if (abs(real_minus_one) < kEpsilon<Scalar>) {
189  halftheta_by_tan_of_halftheta =
190  Scalar(1.) - Scalar(1. / 12) * theta[0] * theta[0];
191  } else {
192  halftheta_by_tan_of_halftheta = -(halftheta * z.y()) / (real_minus_one);
193  }
194  Eigen::Matrix<Scalar, 2, 2> v_inv;
195  v_inv << halftheta_by_tan_of_halftheta, halftheta, -halftheta,
196  halftheta_by_tan_of_halftheta;
197  return v_inv;
198  }
