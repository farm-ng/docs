22 template <class TScalar>
23 class SplineBasisFunction {
24  public:
25  static Eigen::Matrix<TScalar, 3, 4> c() {
26  Eigen::Matrix<TScalar, 3, 4> c;
27  TScalar const o(0);
28 
29  // clang-format off
30  c << TScalar(5./6), TScalar(3./6), -TScalar(3./6), TScalar(1./6),
31  TScalar(1./6), TScalar(3./6), TScalar(3./6), -TScalar(2./6),
32  o, o, o, TScalar(1./6);
33  // clang-format on
34  return c;
35  }
36 
37  static Eigen::Vector3<TScalar> b(TScalar const& u) {
38  // SOPHUS_ASSERT(u >= TScalar(0), "but %", u);
39  // SOPHUS_ASSERT(u < TScalar(1), "but %", u);
40  TScalar u_square(u * u);
41  return c() * Eigen::Vector4<TScalar>(TScalar(1), u, u_square, u * u_square);
42  }
43 
44  static Eigen::Vector3<TScalar> dtB(TScalar const& u, TScalar const& delta_t) {
45  // SOPHUS_ASSERT(u >= TScalar(0), "but %", u);
46  // SOPHUS_ASSERT(u < TScalar(1), "but %", u);
47  return (TScalar(1) / delta_t) * c() *
48  Eigen::Vector4<TScalar>(
49  TScalar(0), TScalar(1), TScalar(2) * u, TScalar(3) * u * u);
50  }
51 
52  static Eigen::Vector3<TScalar> dt2B(
53  TScalar const& u, TScalar const& delta_t) {
54  // SOPHUS_ASSERT(u >= TScalar(0), "but %", u);
55  // SOPHUS_ASSERT(u < TScalar(1), "but %", u);
56  return (TScalar(1) / (delta_t * delta_t)) * c() *
57  Eigen::Vector4<TScalar>(
58  TScalar(0), TScalar(0), TScalar(2), TScalar(6) * u);
59  }
60 };

62 template <class TLieGroup>
63 class BasisSplineFn {
64  public:
65  using LieGroup = TLieGroup;
66  using TScalar = typename LieGroup::Scalar;
67  using Transformation = typename LieGroup::Transformation;
68  using Tangent = typename LieGroup::Tangent;
69 
70  static LieGroup parentFromSpline(
71  LieGroup const& parent_ts_control_point,
72  std::tuple<Tangent, Tangent, Tangent> const& control_tagent_vectors,
73  double u) {
74  auto aa = a(control_tagent_vectors, u);
75  return parent_ts_control_point * std::get<0>(aa) * std::get<1>(aa) *
76  std::get<2>(aa);
77  }
78 
79  static Transformation dtParentFromSpline(
80  LieGroup const& parent_ts_control_point,
81  std::tuple<Tangent, Tangent, Tangent> const& control_tagent_vectors,
82  double u,
83  double delta_t) {
84  auto aa = a(control_tagent_vectors, u);
85  auto dt_aa = dtA(aa, control_tagent_vectors, u, delta_t);
86  return parent_ts_control_point.matrix() *
87  ((std::get<0>(dt_aa) * std::get<1>(aa).matrix() *
88  std::get<2>(aa).matrix()) +
89  (std::get<0>(aa).matrix() * std::get<1>(dt_aa) *
90  std::get<2>(aa).matrix()) +
91  (std::get<0>(aa).matrix() * std::get<1>(aa).matrix() *
92  std::get<2>(dt_aa)));
93  }
94 
95  static Transformation dt2ParentFromSpline(
96  LieGroup const& parent_ts_control_point,
97  std::tuple<Tangent, Tangent, Tangent> const& control_tagent_vectors,
98  double u,
99  double delta_t) {
100  using TScalar = typename LieGroup::Scalar;
101  auto aa = a(control_tagent_vectors, u);
102  auto dt_aa = dtA(aa, control_tagent_vectors, u, delta_t);
103  auto dt2_aa = dt2A(aa, dt_aa, control_tagent_vectors, u, delta_t);
104 
105  return parent_ts_control_point.matrix() *
106  ((std::get<0>(dt2_aa) * std::get<1>(aa).matrix() *
107  std::get<2>(aa).matrix()) +
108  (std::get<0>(aa).matrix() * std::get<1>(dt2_aa) *
109  std::get<2>(aa).matrix()) +
110  (std::get<0>(aa).matrix() * std::get<1>(aa).matrix() *
111  std::get<2>(dt2_aa)) +
112  TScalar(2) * ((std::get<0>(dt_aa) * std::get<1>(dt_aa) *
113  std::get<2>(aa).matrix()) +
114  (std::get<0>(dt_aa) * std::get<1>(aa).matrix() *
115  std::get<2>(dt_aa)) +
116  (std::get<0>(aa).matrix() * std::get<1>(dt_aa) *
117  std::get<2>(dt_aa))));
118  }
119 
120  private:
121  static std::tuple<LieGroup, LieGroup, LieGroup> a(
122  std::tuple<Tangent, Tangent, Tangent> const& control_tagent_vectors,
123  double u) {
124  Eigen::Vector3d b = SplineBasisFunction<double>::b(u);
125  return std::make_tuple(
126  LieGroup::exp(b[0] * std::get<0>(control_tagent_vectors)),
127  LieGroup::exp(b[1] * std::get<1>(control_tagent_vectors)),
128  LieGroup::exp(b[2] * std::get<2>(control_tagent_vectors)));
129  }
130 
131  static std::tuple<Transformation, Transformation, Transformation> dtA(
132  std::tuple<LieGroup, LieGroup, LieGroup> const& aa,
133  std::tuple<Tangent, Tangent, Tangent> const& control_tagent_vectors,
134  double u,
135  double delta_t) {
136  Eigen::Vector3d dt_b = SplineBasisFunction<double>::dtB(u, delta_t);
137  return std::make_tuple(
138  dt_b[0] * std::get<0>(aa).matrix() *
139  LieGroup::hat(std::get<0>(control_tagent_vectors)),
140  dt_b[1] * std::get<1>(aa).matrix() *
141  LieGroup::hat(std::get<1>(control_tagent_vectors)),
142  dt_b[2] * std::get<2>(aa).matrix() *
143  LieGroup::hat(std::get<2>(control_tagent_vectors)));
144  }
145 
146  static std::tuple<Transformation, Transformation, Transformation> dt2A(
147  std::tuple<LieGroup, LieGroup, LieGroup> const& aa,
148  std::tuple<Transformation, Transformation, Transformation> const& dt_aa,
149  std::tuple<Tangent, Tangent, Tangent> const& control_tagent_vectors,
150  double u,
151  double delta_t) {
152  Eigen::Vector3d dt_b = SplineBasisFunction<double>::dtB(u, delta_t);
153  Eigen::Vector3d dt2_b = SplineBasisFunction<double>::dt2B(u, delta_t);
154 
155  return std::make_tuple(
156  (dt_b[0] * std::get<0>(dt_aa).matrix() +
157  dt2_b[0] * std::get<0>(aa).matrix()) *
158  LieGroup::hat(std::get<0>(control_tagent_vectors)),
159  (dt_b[1] * std::get<1>(dt_aa).matrix() +
160  dt2_b[1] * std::get<1>(aa).matrix()) *
161  LieGroup::hat(std::get<1>(control_tagent_vectors)),
162  (dt_b[2] * std::get<2>(dt_aa).matrix() +
163  dt2_b[2] * std::get<2>(aa).matrix()) *
164  LieGroup::hat(std::get<2>(control_tagent_vectors)));
165  }
166 };

438 template <class TLieGroup>
439 class BasisSpline {
440  public:
441  using LieGroup = TLieGroup;
442  using Scalar = typename LieGroup::Scalar;
443  using Transformation = typename LieGroup::Transformation;
444  using Tangent = typename LieGroup::Tangent;
445 
446  BasisSpline(
447  std::vector<LieGroup> parent_ts_control_point, double t0, double delta_t)
448  : impl_(std::move(parent_ts_control_point), delta_t), t0_(t0) {}
449 
450  [[nodiscard]] LieGroup parentFromSpline(double t) const {
451  IndexAndU index_and_u = this->indexAndU(t);
452 
453  return impl_.parentFromSpline(index_and_u.i, index_and_u.u);
454  }
455 
456  [[nodiscard]] Transformation dtParentFromSpline(double t) const {
457  IndexAndU index_and_u = this->indexAndU(t);
458  return impl_.dtParentFromSpline(index_and_u.i, index_and_u.u);
459  }
460 
461  [[nodiscard]] Transformation dt2ParentFromSpline(double t) const {
462  IndexAndU index_and_u = this->indexAndU(t);
463  return impl_.dt2ParentFromSpline(index_and_u.i, index_and_u.u);
464  }
465 
466  [[nodiscard]] double t0() const { return t0_; }
467 
468  [[nodiscard]] double tmax() const {
469  return t0_ + impl_.deltaT() * getNumSegments();
470  }
471 
472  [[nodiscard]] std::vector<LieGroup> const& parentFromsControlPoint() const {
473  return impl_.parentFromsControlPoint();
474  }
475 
476  std::vector<LieGroup>& parentFromsControlPoint() {
477  return impl_.parentFromsControlPoint();
478  }
479 
480  [[nodiscard]] int getNumSegments() const { return impl_.getNumSegments(); }
481 
482  [[nodiscard]] double s(double t) const { return (t - t0_) / impl_.deltaT(); }
483 
484  [[nodiscard]] double deltaT() const { return impl_.deltaT(); }
485 
486  [[nodiscard]] IndexAndU indexAndU(double t) const {
487  SOPHUS_ASSERT(t >= t0_, "{} vs. {}", t, t0_);
488  SOPHUS_ASSERT(t <= this->tmax(), "{} vs. {}", t, this->tmax());
489 
490  double s = this->s(t);
491  double i = NAN;
492  IndexAndU index_and_u;
493  index_and_u.u = std::modf(s, &i);
494  index_and_u.i = int(i);
495  if (index_and_u.u > sophus::kEpsilonF64) {
496  return index_and_u;
497  }
498 
499  // u ~=~ 0.0
500  if (t < 0.5 * this->tmax()) {
501  // First half of spline, keep as is (i, 0.0).
502  return index_and_u;
503  }
504  // Second half of spline, use (i-1, 1.0) instead. This way we can represent
505  // t == tmax (and not just t<tmax).
506  index_and_u.u += 1.0;
507  --index_and_u.i;
508 
509  return index_and_u;
510  }
511 
512  private:
513  BasisSplineImpl<LieGroup> impl_;
514 
515  double t0_;
516 };
