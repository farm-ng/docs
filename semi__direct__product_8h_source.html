71  static auto exp(Eigen::Vector<Scalar, kDof> tangent)
72  -> Eigen::Vector<Scalar, kNumParams> {
73  Eigen::Vector<Scalar, SubGroup::kNumParams> subgroup_params =
74  SubGroup::exp(subgroupTangent(tangent));
75  return params(
76  subgroup_params,
77  (SubGroup::matV(subgroup_params, subgroupTangent(tangent)) *
78  translationTangent(tangent))
79  .eval());
80  }
81 
82  static auto log(Eigen::Vector<Scalar, kNumParams> const& params)
83  -> Eigen::Vector<Scalar, kDof> {
84  Eigen::Vector<Scalar, SubGroup::kDof> subgroup_tangent =
85  SubGroup::log(subgroupParams(params));
86  return tangent(
87  SubGroup::matVInverse(subgroupParams(params), subgroup_tangent) *
88  translation(params),
89  subgroup_tangent);
90  }

92  static auto hat(Eigen::Vector<Scalar, kDof> const& tangent)
93  -> Eigen::Matrix<Scalar, kAmbientDim, kAmbientDim> {
94  Eigen::Matrix<Scalar, kAmbientDim, kAmbientDim> hat_mat;
95  hat_mat.setZero();
96  hat_mat.template topLeftCorner<kPointDim, kPointDim>() =
97  SubGroup::hat(subgroupTangent(tangent));
98  hat_mat.template topRightCorner<kPointDim, 1>() =
99  translationTangent(tangent);
100  return hat_mat;
101  }
102 
103  static auto vee(Eigen::Matrix<Scalar, kAmbientDim, kAmbientDim> const& mat)
104  -> Eigen::Matrix<Scalar, kDof, 1> {
105  return tangent(
106  mat.template topRightCorner<kPointDim, 1>().eval(),
107  SubGroup::vee(
108  mat.template topLeftCorner<kPointDim, kPointDim>().eval()));
109  }

111  // group operations
112  static auto multiplication(
113  Eigen::Vector<Scalar, kNumParams> const& lhs_params,
114  Eigen::Vector<Scalar, kNumParams> const& rhs_params)
115  -> Eigen::Vector<Scalar, kNumParams> {
116  return SemiDirectProductWithTranslation::params(
117  SubGroup::multiplication(
118  subgroupParams(lhs_params), subgroupParams(rhs_params)),
119  SubGroup::action(subgroupParams(lhs_params), translation(rhs_params)) +
120  translation(lhs_params));
121  }
122 
123  static auto inverse(Eigen::Vector<Scalar, kNumParams> const& params)
124  -> Eigen::Vector<Scalar, kNumParams> {
125  Eigen::Vector<Scalar, SubGroup::kNumParams> subgroup_params =
126  SubGroup::inverse(subgroupParams(params));
127  return SemiDirectProductWithTranslation::params(
128  subgroup_params,
129  (-SubGroup::action(subgroup_params, translation(params))).eval());
130  }

175  // derivatives
176  static auto adj(Eigen::Vector<Scalar, kNumParams> const& params)
177  -> Eigen::Matrix<Scalar, kDof, kDof> {
178  Eigen::Matrix<Scalar, kDof, kDof> mat_adjoint;
179 
180  Eigen::Vector<Scalar, SubGroup::kNumParams> subgroup_params =
181  subgroupParams(params);
182 
183  mat_adjoint.template topLeftCorner<kPointDim, kPointDim>() =
184  SubGroup::matrix(subgroup_params);
185  mat_adjoint.template topRightCorner<kPointDim, SubGroup::kDof>() =
186  SubGroup::topRightAdj(subgroup_params, translation(params));
187 
188  mat_adjoint.template bottomLeftCorner<SubGroup::kDof, kPointDim>()
189  .setZero();
190  mat_adjoint.template bottomRightCorner<SubGroup::kDof, SubGroup::kDof>() =
191  SubGroup::adj(subgroup_params);
192 
193  return mat_adjoint;
194  }
