38  static auto areParamsValid(
39  Eigen::Vector<Scalar, kNumParams> const& scale_factors)
40  -> sophus::Expected<Success> {
41  static const Scalar kThr = kEpsilon<Scalar>;
42 
43  if (!(scale_factors.array() >= kThr).all()) {
44  return SOPHUS_UNEXPECTED(
45  "scale factors ({}) not positive.\n",
46  "thr: {}",
47  scale_factors.transpose(),
48  kThr);
49  }
50  return sophus::Expected<Success>{};
51  }

67  static auto hat(Eigen::Vector<Scalar, kDof> const& scale_factors)
68  -> Eigen::Matrix<Scalar, kAmbientDim, kAmbientDim> {
69  Eigen::Matrix<Scalar, kAmbientDim, kAmbientDim> mat;
70  mat.setZero();
71  for (int i = 0; i < kDof; ++i) {
72  mat.diagonal()[i] = scale_factors[i];
73  }

141  static auto matV(
142  Eigen::Vector<Scalar, kNumParams> const& params,
143  Eigen::Vector<Scalar, kDof> const& tangent)
144  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
145  Eigen::Matrix<Scalar, kPointDim, kPointDim> mat =
146  Eigen::Matrix<Scalar, kPointDim, kPointDim>::Identity();
147  for (int i = 0; i < kDof; ++i) {
148  mat(i, i) = (params[i] - 1.0) / tangent[i];
149  }
150  return mat;
151  }

153  static auto matVInverse(
154  Eigen::Vector<Scalar, kNumParams> const& params,
155  Eigen::Vector<Scalar, kNumParams> const& tangent)
156  -> Eigen::Matrix<Scalar, kPointDim, kPointDim> {
157  Eigen::Matrix<Scalar, kPointDim, kPointDim> mat =
158  Eigen::Matrix<Scalar, kPointDim, kPointDim>::Identity();
159  for (int i = 0; i < kDof; ++i) {
160  mat(i, i) = tangent[i] / (params[i] - 1.0);
161  }
162  return mat;
163  }
