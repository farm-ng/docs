18 /// Type-erased image with shared ownership, and read-only access to pixels.
19 /// Type is nullable.
20 ///
21 template <
22  class TPredicate = AnyImagePredicate,
23  class TAllocator = Eigen::aligned_allocator<uint8_t>>
24 class DynImage : public DynImageView<TPredicate> {
25  public:
26  /// Empty image.
27  DynImage() = default;
28 
29  /// Create type-erased image from Image.
30  ///
31  /// Ownership is shared between DynImage and Image, and hence the
32  /// reference count will be increased by one (unless input is empty).
33  /// By design not "explicit".
34  template <class TPixel>
35  DynImage(Image<TPixel, TAllocator> const& image)
36  : DynImage(
37  image.layout(),
38  PixelFormat::fromTemplate<TPixel>(),
39  image.shared_) {
40  static_assert(TPredicate::template isTypeValid<TPixel>());
41  }
42 
43  /// Create type-erased image from MutImage.
44  /// By design not "explicit".
45  template <class TPixel>
46  DynImage(MutImage<TPixel>&& image)
47  : DynImage(Image<TPixel>(std::move(image))) {
48  static_assert(TPredicate::template isTypeValid<TPixel>());
49  }
50 
51  /// Create type-erased image from MutImage.
52  /// By design not "explicit".
53  DynImage(MutDynImage<TPredicate, TAllocator>&& image)
54  : DynImage(
55  image.layout(), image.pixel_format_, std::move(image.unique_)) {
56  image.unique_.reset();
57  image.setViewToEmpty();
58  }
59 
60  /// Create type-image image from provided size and pixel type.
61  /// Pixel data is left uninitialized
62  DynImage(ImageSize const& size, PixelFormat const& pixel_type)
63  : DynImage(MutDynImage<TPredicate, TAllocator>(size, pixel_type)) {}
64 
65  /// Create type-image image from provided layout and pixel type.
66  /// Pixel data is left uninitialized
67  DynImage(ImageLayout const& layout, PixelFormat const& pixel_type)
68  : DynImage(MutDynImage<TPredicate, TAllocator>(layout, pixel_type)) {}
69 
70  /// Return true is this contains data of type TPixel.
71  template <class TPixel>
72  [[nodiscard]] bool has() const noexcept {
73  PixelFormat expected_type = PixelFormat::fromTemplate<TPixel>();
74  return expected_type == this->pixel_format_;
75  }
76 
77  /// Returns typed image.
78  ///
79  /// Precondition: this->has<TPixel>()
80  template <class TPixel>
81  [[nodiscard]] Image<TPixel, TAllocator> image() const noexcept {
82  if (!this->has<TPixel>()) {
83  PixelFormat expected_type = PixelFormat::fromTemplate<TPixel>();
84 
85  SOPHUS_PANIC(
86  "expected type: {}\n"
87  "actual type: {}",
88  expected_type,
89  this->pixel_format_);
90  }
91 
92  return Image<TPixel, TAllocator>(
93  ImageView<TPixel>(
94  this->layout_, reinterpret_cast<TPixel*>(shared_.get())),
95  shared_);
96  }
97 
98  template <class TPixel>
99  Image<TPixel, TAllocator> reinterpretAs(
100  ImageSize reinterpreted_size) const noexcept {
101  SOPHUS_ASSERT_LE(
102  reinterpreted_size.width * sizeof(TPixel), this->layout().pitch_bytes_);
103  SOPHUS_ASSERT_LE(reinterpreted_size.height, this->height());
104 
105  SOPHUS_UNIMPLEMENTED();
106  }
107 
108  [[nodiscard]] size_t useCount() const { return shared_.use_count(); }
109 
110  protected:
111  // Private constructor mainly available for constructing sub-views
112  DynImage(
113  ImageLayout layout,
114  PixelFormat pixel_type,
115  std::shared_ptr<uint8_t> shared)
116  : DynImageView<TPredicate>(layout, pixel_type, shared.get()),
117  shared_(shared) {}
118 
119  std::shared_ptr<uint8_t> shared_;
120 };
