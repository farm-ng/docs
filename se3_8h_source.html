57 /// Se3 base type - implements Se3 class but is storage agnostic.
58 ///
59 /// SE(3) is the group of rotations and translation in 3d. It is the
60 /// semi-direct product of SO(3) and the 3d Euclidean vector space. The class
61 /// is represented using a composition of So3 for rotation and a one 3-vector
62 /// for translation.
63 ///
64 /// SE(3) is neither compact, nor a commutative group.
65 ///
66 ///
67 /// - 4x4 Eigen::Matrix representation:
68 ///
69 /// ```
70 /// | R t |
71 /// | o 1 |
72 /// ```
73 ///
74 /// where ``R`` is a 3x3 rotation matrix, ``t`` a translation 3-vector and
75 /// ``o`` a 3-column vector of zeros.
76 ///
77 /// - Tangent 6-vector: [upsilon, omega],
78 ///
79 /// where ``upsilon`` is the translational velocity 3-vector and ``omega`` the
80 /// rotational velocity 3-vector.
81 ///
82 /// - Internal 7-representation: [t0, t1, t2, qi0, qi1, qi2, qr],
83 ///
84 /// with ``t0, t1, t2`` are the translational components, and ``qi0, qi1, q2``
85 /// the imaginary vector part and ``qr1`` the real part of a unit-length
86 /// quaternion.
87 ///
88 ///
89 /// See So3 for more details of the rotation representation in 3d.
90 ///
91 template <class TDerived>
92 class Se3Base {
93  public:

134  /// Adjoint transformation
135  ///
136  /// This function return the adjoint transformation ``Ad`` of the group
137  /// element ``A`` such that for all ``x`` it holds that
138  /// ``hat(Ad_A * x) = A * hat(x) A^{-1}``. See hat-operator below.
139  ///
140  SOPHUS_FUNC [[nodiscard]] Adjoint adj() const {
141  Eigen::Matrix3<Scalar> const mat_r = so3().matrix();
142  Adjoint res;
143  res.block(0, 0, 3, 3) = mat_r;
144  res.block(3, 3, 3, 3) = mat_r;
145  res.block(0, 3, 3, 3) = So3<Scalar>::hat(translation()) * mat_r;
146  res.block(3, 0, 3, 3) = Eigen::Matrix3<Scalar>::Zero(3, 3);
147  return res;
148  }

244  /// Returns derivative of log(this^{-1} * x) by x at x=this.
245  ///
246  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kDoF, kNumParameters>
247  dxLogThisInvTimesXAtThis() const {
248  Eigen::Matrix<Scalar, kDoF, kNumParameters> j;
249  j.template block<3, 4>(0, 0).setZero();
250  j.template block<3, 3>(0, 4) = so3().inverse().matrix();
251  j.template block<3, 4>(3, 0) = so3().dxLogThisInvTimesXAtThis();
252  j.template block<3, 3>(3, 4).setZero();
253  return j;

263  /// Logarithmic map
264  ///
265  /// Computes the logarithm, the inverse of the group exponential which maps
266  /// element of the group (rigid body transformations) to elements of the
267  /// tangent space (twist).
268  ///
269  /// To be specific, this function computes ``vee(logmat(.))`` with
270  /// ``logmat(.)`` being the matrix logarithm and ``vee(.)`` the vee-operator
271  /// of SE(3).
272  ///
273  SOPHUS_FUNC [[nodiscard]] Tangent log() const {
274  // For the derivation of the logarithm of SE(3), see
275  // J. Gallier, D. Xu, "Computing exponentials of skew symmetric matrices
276  // and logarithms of orthogonal matrices", IJRA 2002.
277  // https:///pdfs.semanticscholar.org/cfe3/e4b39de63c8cabd89bf3feff7f5449fc981d.pdf
278  // (Sec. 6., pp. 8)
279  using std::abs;
280  using std::cos;
281  using std::sin;
282  Tangent vec_upsilon_omega;
283  auto omega_and_theta = so3().logAndTheta();
284  Scalar theta = omega_and_theta.theta;
285  Eigen::Vector3<Scalar> const& vec_omega = omega_and_theta.tangent;
286  vec_upsilon_omega.template tail<3>() = vec_omega;
287  Eigen::Matrix3<Scalar> mat_v_inv =
288  So3<Scalar>::leftJacobianInverse(vec_omega, theta);
289  vec_upsilon_omega.template head<3>() = mat_v_inv * translation();
290  return vec_upsilon_omega;
291  }

348  /// Group action on 3-points.
349  ///
350  /// This function rotates and translates a three dimensional point ``p`` by
351  /// the SE(3) element ``bar_transform_foo = (bar_R_foo, t_bar)`` (= rigid body
352  /// transformation):
353  ///
354  /// ``p_bar = bar_R_foo * p_foo + t_bar``.
355  ///
356  template <
357  typename TPointDerived,
358  typename = typename std::enable_if<
359  IsFixedSizeVector<TPointDerived, 3>::value>::type>
360  SOPHUS_FUNC PointProduct<TPointDerived> operator*(
361  Eigen::MatrixBase<TPointDerived> const& p) const {
362  return so3() * p + translation();

366  ///
367  template <
368  typename THPointDerived,

623  Scalar const theta_sq = vec_omega.squaredNorm();
624  Eigen::Matrix3<Scalar> const mat_upsilon = So3<Scalar>::hat(vec_upsilon);
625 
626  Eigen::Matrix3<Scalar> q;
627  if (theta_sq < kEpsilon<Scalar> * kEpsilon<Scalar>) {
628  q = half * mat_upsilon;
629 
630  } else {
631  Scalar const theta = sqrt(theta_sq);
632  Scalar const i_theta = Scalar(1) / theta;
633  Scalar const i_theta_sq = i_theta * i_theta;
634  Scalar const i_theta_po4 = i_theta_sq * i_theta_sq;
635  Scalar const st = sin(theta);
636  Scalar const ct = cos(theta);
637  Scalar const c1 = i_theta_sq - st * i_theta_sq * i_theta;
638  Scalar const c2 = half * i_theta_sq + ct * i_theta_po4 - i_theta_po4;
639  Scalar const c3 = i_theta_po4 + half * ct * i_theta_po4 -
640  Scalar(1.5) * st * i_theta * i_theta_po4;
641 
642  Eigen::Matrix3<Scalar> const mat_omega = So3<Scalar>::hat(vec_omega);
643  Eigen::Matrix3<Scalar> const mat_omega_upsilon = mat_omega * mat_upsilon;
644  Eigen::Matrix3<Scalar> const mat_omega_upsilon_omega =
645  mat_omega_upsilon * mat_omega;
646  q = half * mat_upsilon +
647  c1 * (mat_omega_upsilon + mat_upsilon * mat_omega +
648  mat_omega_upsilon_omega) -
649  c2 * (theta_sq * mat_upsilon + Scalar(2) * mat_omega_upsilon_omega) +
650  c3 * (mat_omega_upsilon_omega * mat_omega +
651  mat_omega * mat_omega_upsilon_omega);
652  }
653  return q;
654  }

985  /// hat-operator
986  ///
987  /// It takes in the 6-vector representation (= twist) and returns the
988  /// corresponding matrix representation of Lie algebra element.
989  ///
990  /// Formally, the hat()-operator of SE(3) is defined as
991  ///
992  /// ``hat(.): R^6 -> R^{4x4}, hat(a) = sum_i a_i * G_i`` (for i=0,...,5)
993  ///
994  /// with ``G_i`` being the ith infinitesimal generator of SE(3).
995  ///
996  /// The corresponding inverse is the vee()-operator, see below.
997  ///
998  SOPHUS_FUNC static Transformation hat(Tangent const& a) {
999  Transformation omega;
1000  omega.setZero();
1001  omega.template topLeftCorner<3, 3>() =
1002  So3<Scalar>::hat(a.template tail<3>());
1003  omega.col(3).template head<3>() = a.template head<3>();
1004  return omega;
1005  }

1007  /// Lie bracket
1008  ///
1009  /// It computes the Lie bracket of SE(3). To be more specific, it computes
1010  ///
1011  /// ``[omega_1, omega_2]_se3 := vee([hat(omega_1), hat(omega_2)])``
1012  ///
1013  /// with ``[A,B] := AB-BA`` being the matrix commutator, ``hat(.)`` the
1014  /// hat()-operator and ``vee(.)`` the vee()-operator of SE(3).
1015  ///
1016  SOPHUS_FUNC static Tangent lieBracket(Tangent const& a, Tangent const& b) {
1017  Eigen::Vector3<Scalar> const upsilon1 = a.template head<3>();
1018  Eigen::Vector3<Scalar> const upsilon2 = b.template head<3>();
1019  Eigen::Vector3<Scalar> const omega1 = a.template tail<3>();
1020  Eigen::Vector3<Scalar> const omega2 = b.template tail<3>();
1021 
1022  Tangent res;
1023  res.template head<3>() = omega1.cross(upsilon2) + upsilon1.cross(omega2);
1024  res.template tail<3>() = omega1.cross(omega2);
1025 
1026  return res;
1027  }

1089  /// vee-operator
1090  ///
1091  /// It takes 4x4-matrix representation ``Omega`` and maps it to the
1092  /// corresponding 6-vector representation of Lie algebra.
1093  ///
1094  /// This is the inverse of the hat()-operator, see above.
1095  ///
1096  /// Precondition: ``Omega`` must have the following structure:
1097  ///
1098  /// | 0 -f e a |
1099  /// | f 0 -d b |
1100  /// | -e d 0 c
1101  /// | 0 0 0 0 | .
1102  ///
1103  SOPHUS_FUNC static Tangent vee(Transformation const& omega) {
1104  Tangent upsilon_omega;
1105  upsilon_omega.template head<3>() = omega.col(3).template head<3>();
1106  upsilon_omega.template tail<3>() =
1107  So3<Scalar>::vee(omega.template topLeftCorner<3, 3>());
1108  return upsilon_omega;
