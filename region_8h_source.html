43 /// A region is a closed interval [a, b] with a being the lower bound (=min) and
44 /// b being the upper bound (=max).
45 ///
46 /// Here, the bounds a, b sre either boths scalars (e.g. floats, doubles)
47 /// or fixed length vectors/arrays (such as Eigen::Vector3f or Eigen::Array2d).
48 ///
49 /// Special case for integer numbers:
50 ///
51 /// An integer number X is considered being equivalent to a real region
52 /// [X-0.5, X+0.5].
53 template <PointType TPoint>
54 class Region {
55  public:
56  static bool constexpr kIsInteger = PointTraits<TPoint>::kIsInteger;
57  using Point = TPoint;
58  using Scalar = typename PointTraits<TPoint>::Scalar;
59 
60  static int constexpr kRows = PointTraits<TPoint>::kRows;
61  static int constexpr kCols = PointTraits<TPoint>::kCols;
62  static_assert(kRows >= 1);
63  static_assert(kCols >= 1);
64  static int constexpr kDim = kRows * kCols;
65 
66  /// Creates an uninitialized region.
67  static Region<TPoint> uninitialized() noexcept { return Region(UninitTag{}); }
68 
69  /// Creates and empty region.
70  static Region<TPoint> empty() noexcept {
71  auto region = Region<TPoint>::uninitialized();
72  region.min_max_ = {
73  PointTraits<TPoint>::max(), PointTraits<TPoint>::lowest()};
74  return region;
75  }
76 
77  /// Creates unbounded region.
78  ///
79  /// If TPoint is floating point, the region is [-inf, +inf].
80  static Region<TPoint> unbounded() noexcept {
81  auto region = Region<TPoint>::uninitialized();
82  region.min_max_ = {PointTraits<TPoint>::min(), PointTraits<TPoint>::max()};
83  return region;
84  }

86  /// Creates a region from a given point.
87  ///
88  /// If TPoint is a floating point then the region is considered degenerated.
89  static Region<TPoint> from(TPoint const& p) noexcept {
90  auto region = Region<TPoint>::empty();
91  SOPHUS_ASSERT(!isNan(p));
92  region.extend(p);
93  return region;
94  }

96  /// Creates Region from two points, min and max.
97  ///
98  /// The points min, max need not to be ordered. After construction it will be
99  /// true that this->min() <= this->max().
100  static Region<TPoint> fromMinMax(
101  TPoint const& min, TPoint const& max) noexcept {
102  auto region = Region<TPoint>::empty();
103  SOPHUS_ASSERT(!isNan(min));
104  SOPHUS_ASSERT(!isNan(max));
105  region.extend(min);
106  region.extend(max);
107  return region;
108  }
109 
110  /// Convenient constructor to create a Segment from two points p1 and p2.
111  ///
112  /// The points need not to be ordered. After construction it will be true
113  /// that this->min() <= this->max().
114  ///
115  /// Note: This constructor is only available for scalar regions. For
116  /// multi-dim regions, use the fromMinMax() factory instead.,
117  template <class TScalar>
118  requires std::is_same_v<TPoint, TScalar> Region(
119  TScalar const& p1, TScalar const& p2)
120  noexcept : min_max_{p1, p1} {
121  SOPHUS_ASSERT(!isNan(p1));
122  SOPHUS_ASSERT(!isNan(p2));
123  this->extend(p2);
124  }

173  /// Returns the lower bound of the region.
174  [[nodiscard]] TPoint const& min() const noexcept { return min_max_[0]; }
175 
176  /// Returns the upper bound of the region.
177  [[nodiscard]] TPoint const& max() const noexcept { return min_max_[1]; }
178 
179  /// Returns the clamped version of the given point.
180  [[nodiscard]] TPoint clamp(TPoint const& point) const noexcept {
181  return sophus::clamp(point, min_max_[0], min_max_[1]);
182  }
183 
184  /// Returns true if the region contains the given point.
185  [[nodiscard]] bool contains(TPoint const& point) const noexcept {
186  return isLessEqual(min(), point) && isLessEqual(point, max());
187  }
188 
189  /// Returns the range of the region.
190  ///
191  /// It is zero if the region is not proper.
192  [[nodiscard]] TPoint range() const noexcept {
193  if constexpr (kIsInteger) {
194  // For integers, we consider e.g. {2} == [1.5, 2.5] hence range of
195  // Region(2, 3) == [1.5, 3.5] is 2.
196  return plus(eval(max() - min()), 1);
197  }
198  return eval(max() - min());
199  }
200 
201  /// Returns the mid point.
202  ///
203  /// Note: If TPoint is an integer point then the result will be rounded to
204  /// the closed integer.
205  [[nodiscard]] TPoint mid() const noexcept {
206  return eval(min() + range() / 2);
207  }
208 
209  /// Extends this by other region.
210  Region<TPoint>& extend(Region const& other) noexcept {
211  min_max_[0] = sophus::min(min(), other.min());
212  min_max_[1] = sophus::max(max(), other.max());
213  return *this;
214  }
215 
216  /// Extends this by given point.
217  Region<TPoint>& extend(TPoint const& point) noexcept {
218  min_max_[0] = sophus::min(min(), point);
219  min_max_[1] = sophus::max(max(), point);
220  return *this;
221  }
222 
223  /// Returns translated region.
224  [[nodiscard]] Region<TPoint> translated(TPoint const& p) const noexcept {
225  return Region<TPoint>::fromMinMax(min_max_[0] + p, min_max_[1] + p);
226  }

228  template <class TOtherPoint>
229  Region<TOtherPoint> cast() const noexcept {
230  if (isEmpty()) {
231  return Region<TOtherPoint>::empty();
232  }
233  if constexpr (kIsInteger == Region<TOtherPoint>::kIsInteger) {
234  // case 1: floating => floating and integer => integer is trivial
235  return Region<TOtherPoint>(
236  sophus::cast<TOtherPoint>(min()), sophus::cast<TOtherPoint>(max()));
237  }
238  if constexpr (kIsInteger && !Region<TOtherPoint>::kIsInteger) {
239  // case 2: integer to floating.
240  //
241  // example: [2, 5] -> [1.5, 5.5]
242  return Region<TOtherPoint>(
243  plus(sophus::cast<TOtherPoint>(min()), -0.5),
244  plus(sophus::cast<TOtherPoint>(max()), 0.5));
245  }
246  // case 3: floating to integer.
247  static_assert(
248  kIsInteger || !Region<TOtherPoint>::kIsInteger,
249  "For floating to integer: call encloseCast() or roundCast() "
250  "instead.");
251  }

253  /// Returns the smallest integer region which contains this.
254  ///
255  /// example: [1.2, 1.3] -> [1, 2]
256  template <class TOtherPoint>
257  Region<TOtherPoint> encloseCast() const noexcept {
258  static_assert(!kIsInteger && Region<TOtherPoint>::kIsInteger);
259  return Region<TOtherPoint>(
260  sophus::cast<TOtherPoint>(sophus::floor(min())),
261  sophus::cast<TOtherPoint>(sophus::ceil(max())));
262  }
263 
264  /// Rounds given region bounds and returns resulting integer region.
265  ///
266  /// example: [1.2, 2.3] -> [1, 2]
267  /// example: [1.1, 2.7] -> [1, 3]
268  template <class TOtherPoint>
269  Region<TOtherPoint> roundCast() const noexcept {
270  static_assert(!kIsInteger && Region<TOtherPoint>::kIsInteger);
271  return Region<TOtherPoint>(
272  sophus::cast<TOtherPoint>(sophus::round(min())),
273  sophus::cast<TOtherPoint>(sophus::round(max())));
274  }
275 
276  /// Returns true if region is empty.
277  [[nodiscard]] bool isEmpty() const noexcept {
278  return allTrue(this->min() == PointTraits<TPoint>::max()) ||
279  allTrue(this->max() == PointTraits<TPoint>::lowest());
280  }
281 
282  /// Returns true if region contains a single floating point number.
283  [[nodiscard]] bool isDegenerated() const noexcept {
284  return !kIsInteger && allTrue(min() == max());
285  }
286 
287  /// Returns true if region is neither empty nor degenerated.
288  /// Hence it contains a range of values.
289  [[nodiscard]] bool isProper() const noexcept {
290  return !this->isEmpty() && !this->isDegenerated();
291  }
292 
293  /// Returns true if region has no bounds.
294  [[nodiscard]] bool isUnbounded() const noexcept {
295  return allTrue(this->min() == PointTraits<TPoint>::min()) &&
296  allTrue(this->max() == PointTraits<TPoint>::max());
297  }
298 
299  private:
300  explicit Region(UninitTag /*unused*/) {}
301 
302  // invariant: this->isEmpty() or min_max[0] <= min_max[1]
303  std::array<TPoint, 2> min_max_;
304 };
