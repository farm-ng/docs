51 /// Bins pixel down, factor of 0.5.
52 ///
53 /// See for details:
54 /// https://docs.google.com/document/d/1xmhCMWklP2UoQMGaMqFnsoPWoeMvBfXN7S8-ie9k0UA/edit#heading=h.elfm6123mecj
55 template <class TScalar>
56 Eigen::Matrix<TScalar, 2, 1> binDown(Eigen::Matrix<TScalar, 2, 1> const& in) {
57  Eigen::Matrix<TScalar, 2, 1> out;
58  // Map left image border from -0.5 to 0.0, then scale down, then
59  // map the left image border from 0.0 back to -0.5
60  out[0] = TScalar(0.5) * (in[0] + 0.5) - 0.5; // cx
61  out[1] = TScalar(0.5) * (in[1] + 0.5) - 0.5; // cy
62  return out;
63 }
64 
65 /// Bins pixel up, factor of 2.0.
66 ///
67 /// See for details:
68 /// https://docs.google.com/document/d/1xmhCMWklP2UoQMGaMqFnsoPWoeMvBfXN7S8-ie9k0UA/edit#heading=h.elfm6123mecj
69 template <class TScalar>
70 Eigen::Matrix<TScalar, 2, 1> binUp(Eigen::Matrix<TScalar, 2, 1> const& in) {
71  Eigen::Matrix<TScalar, 2, 1> out;
72  // Map left image border from -0.5 to 0.0, then scale up, then
73  // map the left image border from 0.0 back to -0.5
74  out[0] = TScalar(2.0) * (in[0] + 0.5) - 0.5; // cx
75  out[1] = TScalar(2.0) * (in[1] + 0.5) - 0.5; // cy
76  return out;
77 }

147  /// Subsamples pixel down, factor of 0.5.
148  ///
149  /// See for details:
150  /// https://docs.google.com/document/d/1xmhCMWklP2UoQMGaMqFnsoPWoeMvBfXN7S8-ie9k0UA/edit#heading=h.97r8rr8owwpc
151  ///
152  /// If the original width [height] is odd, the new width [height] will be:
153  /// (width+1)/2 [height+1)/2].
154  [[nodiscard]] CameraModelT subsampleDown() const {
155  Params params = this->params_;
156  params[0] = TScalar(0.5) * params[0]; // fx
157  params[1] = TScalar(0.5) * params[1]; // fy
158  params.template segment<2>(2) = ::sophus::subsampleDown(
159  params.template segment<2>(2).eval()); // cx, cy

161  }

207  [[nodiscard]] CameraModelT scale(ImageSize const& image_size) const {
208  Params params = this->params_;
209  params[0] = TScalar(image_size.width) / TScalar(image_size_.width) *
210  params[0]; // fx
211  params[1] = TScalar(image_size.height) / TScalar(image_size_.height) *
212  params[1]; // fy
213  params[2] = TScalar(image_size.width) / TScalar(image_size_.width) *
214  params[2]; // cx
215  params[3] = TScalar(image_size.height) / TScalar(image_size_.height) *
216  params[3]; // cy
217  return CameraModelT({image_size.width, image_size.height}, params);
218  }

248  for (int v = 0; v < table.height(); ++v) {
249  Eigen::Vector2f* row_ptr = table.rowPtrMut(v);
