64 /// So3 base type - implements So3 class but is storage agnostic.
65 ///
66 /// SO(3) is the group of rotations in 3d. As a matrix group, it is the set of
67 /// matrices which are orthogonal such that ``R * R' = I`` (with ``R'`` being
68 /// the transpose of ``R``) and have a positive determinant. In particular, the
69 /// determinant is 1. Internally, the group is represented as a unit quaternion.
70 /// Unit quaternion can be seen as members of the special unitary group SU(2).
71 /// SU(2) is a double cover of SO(3). Hence, for every rotation matrix ``R``,
72 /// there exist two unit quaternions: ``(r, v)`` and ``(-r, -v)``, with ``r``
73 /// the real part and ``v`` being the imaginary 3-vector part of the quaternion.
74 ///
75 /// SO(3) is a compact, but non-commutative group. First it is compact since the
76 /// set of rotation matrices is a closed and bounded set. Second it is
77 /// non-commutative since the equation ``R_1 * R_2 = R_2 * R_1`` does not hold
78 /// in general. For example rotating an object by some degrees about its
79 /// ``x``-axis and then by some degrees about its y axis, does not lead to the
80 /// same orientation when rotation first about ``y`` and then about ``x``.
81 ///
82 /// Class invariant: The 2-norm of ``unit_quaternion`` must be close to 1.
83 /// Technically speaking, it must hold that:
84 ///
85 /// ``|unit_quaternion().squaredNorm() - 1| <= Constants::epsilon()``.
86 template <class TDerived>
87 class So3Base {

208  /// Returns derivative of this * So3::exp(x) wrt. x at x=0.
209  ///
210  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kNumParameters, kDoF>
211  dxThisMulExpXAt0() const {
212  Eigen::Matrix<Scalar, kNumParameters, kDoF> j;
213  Eigen::Quaternion<Scalar> const q = unitQuaternion();
214  Scalar const c0 = Scalar(0.5) * q.w();
215  Scalar const c1 = Scalar(0.5) * q.z();
216  Scalar const c2 = -c1;
217  Scalar const c3 = Scalar(0.5) * q.y();
218  Scalar const c4 = Scalar(0.5) * q.x();
219  Scalar const c5 = -c4;
220  Scalar const c6 = -c3;
221  j(0, 0) = c0;
222  j(0, 1) = c2;
223  j(0, 2) = c3;
224  j(1, 0) = c1;
225  j(1, 1) = c0;
226  j(1, 2) = c5;
227  j(2, 0) = c6;
228  j(2, 1) = c4;
229  j(2, 2) = c0;
230  j(3, 0) = c5;
231  j(3, 1) = c6;
232  j(3, 2) = c2;
233 
234  return j;
235  }

237  /// Returns derivative of log(this^{-1} * x) wrt. x at x=this.
238  ///
239  SOPHUS_FUNC [[nodiscard]] Eigen::Matrix<Scalar, kDoF, kNumParameters>
240  dxLogThisInvTimesXAtThis() const {
241  auto& q = unitQuaternion();
242  Eigen::Matrix<Scalar, kDoF, kNumParameters> j;
243  // clang-format off
244  j << q.w(), q.z(), -q.y(), -q.x(),
245  -q.z(), q.w(), q.x(), -q.y(),
246  q.y(), -q.x(), q.w(), -q.z();
247  // clang-format on
248  return j * Scalar(2.);
249  }

281  /// As above, but also returns ``theta = |omega|``.
282  ///
283  SOPHUS_FUNC [[nodiscard]] TangentAndTheta logAndTheta() const {
284  TangentAndTheta j;
285  using std::abs;
286  using std::atan2;
287  using std::sqrt;
288  Scalar squared_n = unitQuaternion().vec().squaredNorm();
289  Scalar w = unitQuaternion().w();
290 
291  Scalar two_atan_nbyw_by_n;
292 
293  /// Atan-based log thanks to
294  ///
295  /// C. Hertzberg et al.:
296  /// "Integrating Generic Sensor Fusion Algorithms with Sound State
297  /// Representation through Encapsulation of Manifolds"
298  /// Information Fusion, 2011
299 
300  if (squared_n < kEpsilon<Scalar> * kEpsilon<Scalar>) {
301  // If quaternion is normalized and n=0, then w should be 1;
302  // w=0 should never happen here!
303  SOPHUS_ASSERT(
304  abs(w) >= kEpsilon<Scalar>,
305  "Quaternion ({}) should be normalized!",
306  unitQuaternion().coeffs().transpose().eval());
307  Scalar squared_w = w * w;
308  two_atan_nbyw_by_n =
309  Scalar(2) / w - Scalar(2.0 / 3.0) * (squared_n) / (w * squared_w);
310  j.theta = Scalar(2) * squared_n / w;
311  } else {
312  Scalar n = sqrt(squared_n);
313 
314  // w < 0 ==> cos(theta/2) < 0 ==> theta > pi
315  //
316  // By convention, the condition |theta| < pi is imposed by wrapping theta
317  // to pi; The wrap operation can be folded inside evaluation of atan2
318  //
319  // theta - pi = atan(sin(theta - pi), cos(theta - pi))
320  // = atan(-sin(theta), -cos(theta))
321  //
322  Scalar atan_nbyw =
323  (w < Scalar(0)) ? Scalar(atan2(-n, -w)) : Scalar(atan2(n, w));
324  two_atan_nbyw_by_n = Scalar(2) * atan_nbyw / n;
325  j.theta = two_atan_nbyw_by_n * n;
326  }
327 
328  j.tangent = two_atan_nbyw_by_n * unitQuaternion().vec();
329  return j;
330  }

364  template <
365  typename TQuaternionProductType,
366  typename TQuaternionTypeA,
367  typename TQuaternionTypeB>
368  static TQuaternionProductType quaternionProduct(
369  TQuaternionTypeA const& a, TQuaternionTypeB const& b) {
370  return TQuaternionProductType(
371  a.w() * b.w() - a.x() * b.x() - a.y() * b.y() - a.z() * b.z(),
372  a.w() * b.x() + a.x() * b.w() + a.y() * b.z() - a.z() * b.y(),
373  a.w() * b.y() + a.y() * b.w() + a.z() * b.x() - a.x() * b.z(),
374  a.w() * b.z() + a.z() * b.w() + a.x() * b.y() - a.y() * b.x());
375  }

379  template <class TOtherDerived>
380  SOPHUS_FUNC So3Product<TOtherDerived> operator*(

393  /// Group action on 3-points.
394  ///
395  /// This function rotates a 3 dimensional point ``p`` by the So3 element
396  /// ``bar_R_foo`` (= rotation matrix): ``p_bar = bar_R_foo * p_foo``.
397  ///
398  /// Since So3 is internally represented by a unit quaternion ``q``, it is
399  /// implemented as ``p_bar = q * p_foo * q^{*}``
400  /// with ``q^{*}`` being the quaternion conjugate of ``q``.
401  ///
402  /// Geometrically, ``p`` is rotated by angle ``|omega|`` around the
403  /// axis ``omega/|omega|`` with ``omega := vee(log(bar_R_foo))``.
404  ///
405  /// For ``vee``-operator, see below.
406  ///
407  template <
408  typename TPointDerived,
409  typename = typename std::enable_if<
410  IsFixedSizeVector<TPointDerived, 3>::value>::type>
411  SOPHUS_FUNC PointProduct<TPointDerived> operator*(
412  Eigen::MatrixBase<TPointDerived> const& p) const {
413  /// NOTE: We cannot use Eigen's Quaternion transformVector because it always
414  /// returns aEigen::Vector3 of the same Scalar as this quaternion, so it is
415  /// not able to be applied to Jets and doubles correctly.
416  QuaternionType const& q = unitQuaternion();
417  PointProduct<TPointDerived> uv = q.vec().cross(p);
418  uv += uv;
419  return p + q.w() * uv + q.vec().cross(uv);
420  }
421 
422  /// Group action on homogeneous 3-points. See above for more details.
423  template <
424  typename THPointDerived,
425  typename = typename std::enable_if<
426  IsFixedSizeVector<THPointDerived, 4>::value>::type>
427  SOPHUS_FUNC HomogeneousPointProduct<THPointDerived> operator*(
428  Eigen::MatrixBase<THPointDerived> const& p) const {
429  auto const rp = *this * p.template head<3>();
430  return HomogeneousPointProduct<THPointDerived>(rp(0), rp(1), rp(2), p(3));
431  }

575  /// Returns the left Jacobian on lie group. See 1st entry in rightmost column
576  /// in: http://asrl.utias.utoronto.ca/~tdb/bib/barfoot_ser17_identities.pdf
577  ///
578  /// A precomputed `theta` can be optionally passed in
579  ///
580  /// Warning: Not to be confused with dxExpX(), which is derivative of the
581  /// internal quaternion representation of So3 wrt the tangent vector
582  ///
583  SOPHUS_FUNC static Eigen::Matrix<Scalar, kDoF, kDoF> leftJacobian(
584  Tangent const& omega,
585  std::optional<Scalar> const& maybe_theta = std::nullopt) {
586  using std::cos;
587  using std::sin;
588  using std::sqrt;
589 
590  Scalar const theta_sq =
591  maybe_theta ? *maybe_theta * *maybe_theta : omega.squaredNorm();
592  Eigen::Matrix3<Scalar> const mat_omega = So3<Scalar>::hat(omega);
593  Eigen::Matrix3<Scalar> const mat_omega_sq = mat_omega * mat_omega;
594  Eigen::Matrix3<Scalar> v;
595 
596  if (theta_sq < kEpsilon<Scalar> * kEpsilon<Scalar>) {
597  v = Eigen::Matrix3<Scalar>::Identity() + Scalar(0.5) * mat_omega;
598  } else {
599  Scalar theta = maybe_theta ? *maybe_theta : sqrt(theta_sq);
600  v = Eigen::Matrix3<Scalar>::Identity() +
601  (Scalar(1) - cos(theta)) / theta_sq * mat_omega +
602  (theta - sin(theta)) / (theta_sq * theta) * mat_omega_sq;
603  }
604  return v;
605  }

607  SOPHUS_FUNC static Eigen::Matrix<Scalar, kDoF, kDoF> leftJacobianInverse(
608  Tangent const& omega,
609  std::optional<Scalar> const& maybe_theta = std::nullopt) {
610  using std::cos;
611  using std::sin;
612  using std::sqrt;
613  Scalar const theta_sq =
614  maybe_theta ? *maybe_theta * *maybe_theta : omega.squaredNorm();
615  Eigen::Matrix3<Scalar> const mat_omega = So3<Scalar>::hat(omega);
616 
617  Eigen::Matrix3<Scalar> v_inv;
618  if (theta_sq < kEpsilon<Scalar> * kEpsilon<Scalar>) {
619  v_inv = Eigen::Matrix3<Scalar>::Identity() - Scalar(0.5) * mat_omega +
620  Scalar(1. / 12.) * (mat_omega * mat_omega);
621 
622  } else {
623  Scalar const theta = maybe_theta ? *maybe_theta : sqrt(theta_sq);
624  Scalar const half_theta = Scalar(0.5) * theta;
625 
626  v_inv = Eigen::Matrix3<Scalar>::Identity() - Scalar(0.5) * mat_omega +
627  (Scalar(1) -
628  Scalar(0.5) * theta * cos(half_theta) / sin(half_theta)) /
629  (theta * theta) * (mat_omega * mat_omega);
630  }
631  return v_inv;
632  }

641  /// Returns derivative of exp(x) wrt. x.
642  ///
643  SOPHUS_FUNC static Eigen::Matrix<Scalar, kNumParameters, kDoF> dxExpX(
644  Tangent const& omega) {
645  using std::cos;
646  using std::exp;
647  using std::sin;
648  using std::sqrt;
649  Scalar const c0 = omega[0] * omega[0];
650  Scalar const c1 = omega[1] * omega[1];
651  Scalar const c2 = omega[2] * omega[2];
652  Scalar const c3 = c0 + c1 + c2;
653 
654  if (c3 < kEpsilon<Scalar>) {
655  return dxExpXAt0();
656  }
657 
658  Scalar const c4 = sqrt(c3);
659  Scalar const c5 = 1.0 / c4;
660  Scalar const c6 = 0.5 * c4;
661  Scalar const c7 = sin(c6);
662  Scalar const c8 = c5 * c7;
663  Scalar const c9 = pow(c3, -3.0L / 2.0L);
664  Scalar const c10 = c7 * c9;
665  Scalar const c11 = Scalar(1.0) / c3;
666  Scalar const c12 = cos(c6);
667  Scalar const c13 = Scalar(0.5) * c11 * c12;
668  Scalar const c14 = c7 * c9 * omega[0];
669  Scalar const c15 = Scalar(0.5) * c11 * c12 * omega[0];
670  Scalar const c16 = -c14 * omega[1] + c15 * omega[1];
671  Scalar const c17 = -c14 * omega[2] + c15 * omega[2];
672  Scalar const c18 = omega[1] * omega[2];
673  Scalar const c19 = -c10 * c18 + c13 * c18;
674  Scalar const c20 = Scalar(0.5) * c5 * c7;
675  Eigen::Matrix<Scalar, kNumParameters, kDoF> j;
676  j(0, 0) = -c0 * c10 + c0 * c13 + c8;
677  j(0, 1) = c16;
678  j(0, 2) = c17;
679  j(1, 0) = c16;
680  j(1, 1) = -c1 * c10 + c1 * c13 + c8;
681  j(1, 2) = c19;
682  j(2, 0) = c17;
683  j(2, 1) = c19;
684  j(2, 2) = -c10 * c2 + c13 * c2 + c8;
685  j(3, 0) = -c20 * omega[0];
686  j(3, 1) = -c20 * omega[1];
687  j(3, 2) = -c20 * omega[2];
688  return j;
689  }

719  ///
720  /// This functions takes in an element of tangent space (= rotation vector
721  /// ``omega``) and returns the corresponding element of the group SO(3).
722  ///
723  /// To be more specific, this function computes ``expmat(hat(omega))``
724  /// with ``expmat(.)`` being the matrix exponential and ``hat(.)`` being the
725  /// hat()-operator of SO(3).

731  /// As above, but also returns ``theta = |omega|`` as out-parameter.
732  ///
733  /// Precondition: ``theta`` must not be ``nullptr``.
734  ///
735  SOPHUS_FUNC static So3AndTheta expAndTheta(Tangent const& omega) {
736  So3AndTheta so3_and_theta;
737 
738  using std::abs;
739  using std::cos;
740  using std::sin;
741  using std::sqrt;
742  Scalar theta_sq = omega.squaredNorm();
743 
744  Scalar imag_factor;
745  Scalar real_factor;
746  if (theta_sq < kEpsilon<Scalar> * kEpsilon<Scalar>) {
747  so3_and_theta.theta = Scalar(0);
748  Scalar theta_po4 = theta_sq * theta_sq;
749  imag_factor = Scalar(0.5) - Scalar(1.0 / 48.0) * theta_sq +
750  Scalar(1.0 / 3840.0) * theta_po4;
751  real_factor = Scalar(1) - Scalar(1.0 / 8.0) * theta_sq +
752  Scalar(1.0 / 384.0) * theta_po4;
753  } else {
754  so3_and_theta.theta = sqrt(theta_sq);
755  Scalar half_theta = Scalar(0.5) * (so3_and_theta.theta);
756  Scalar sin_half_theta = sin(half_theta);
757  imag_factor = sin_half_theta / (so3_and_theta.theta);
758  real_factor = cos(half_theta);
759  }
760 
761  So3 so3;
762  so3.mutUnitQuaternion() = QuaternionMember(
763  real_factor,
764  imag_factor * omega.x(),
765  imag_factor * omega.y(),
766  imag_factor * omega.z());
767  so3_and_theta.so3 = so3;
768  SOPHUS_ASSERT(
769  abs(so3_and_theta.so3.unitQuaternion().squaredNorm() - Scalar(1)) <
770  sophus::kEpsilon<Scalar>,
771  "So3::exp failed! omega: {}, real: {}, img: {}",
772  omega.transpose().eval(),
773  real_factor,
774  imag_factor);
775  return so3_and_theta;
776  }

786  /// Returns the ith infinitesimal generators of SO(3).
787  ///
788  /// The infinitesimal generators of SO(3) are:
789  ///
790  /// ```
791  /// | 0 0 0 |
792  /// G_0 = | 0 0 -1 |
793  /// | 0 1 0 |
794  ///
795  /// | 0 0 1 |
796  /// G_1 = | 0 0 0 |
797  /// | -1 0 0 |
798  ///
799  /// | 0 -1 0 |
800  /// G_2 = | 1 0 0 |
801  /// | 0 0 0 |
802  /// ```
803  ///
804  /// Precondition: ``i`` must be 0, 1 or 2.
805  ///
806  SOPHUS_FUNC static Transformation generator(int i) {
807  SOPHUS_ASSERT(i >= 0 && i <= 2, "i should be in range [0,2].");
808  Tangent e;
809  e.setZero();
810  e[i] = Scalar(1);
811  return hat(e);
812  }

814  /// hat-operator
815  ///
816  /// It takes in the 3-vector representation ``omega`` (= rotation vector) and
817  /// returns the corresponding matrix representation of Lie algebra element.
818  ///
819  /// Formally, the hat()-operator of SO(3) is defined as
820  ///
821  /// ``hat(.): R^3 -> R^{3x3}, hat(omega) = sum_i omega_i * G_i``
822  /// (for i=0,1,2)
823  ///
824  /// with ``G_i`` being the ith infinitesimal generator of SO(3).
825  ///
826  /// The corresponding inverse is the vee()-operator, see below.
827  ///
828  SOPHUS_FUNC static Transformation hat(Tangent const& omega) {
829  Transformation mat_omega;
830  // clang-format off
831  mat_omega <<
832  Scalar(0), -omega(2), omega(1),
833  omega(2), Scalar(0), -omega(0),
834  -omega(1), omega(0), Scalar(0);
835  // clang-format on
836  return mat_omega;
837  }
838 
839  /// Lie bracket
840  ///
841  /// It computes the Lie bracket of SO(3). To be more specific, it computes
842  ///
843  /// ``[omega_1, omega_2]_so3 := vee([hat(omega_1), hat(omega_2)])``
844  ///
845  /// with ``[A,B] := AB-BA`` being the matrix commutator, ``hat(.)`` the
846  /// hat()-operator and ``vee(.)`` the vee()-operator of So3.
847  ///
848  /// For the Lie algebra so3, the Lie bracket is simply the cross product:
849  ///
850  /// ``[omega_1, omega_2]_so3 = omega_1 x omega_2.``
851  ///
852  SOPHUS_FUNC static Tangent lieBracket(
853  Tangent const& omega1, Tangent const& omega2) {
854  return omega1.cross(omega2);
855  }

876  /// Based on: http://planning.cs.uiuc.edu/node198.html
877  ///
878  template <class TUniformRandomBitGenerator>

880  static_assert(
881  kIsUniformRandomBitGeneratorV<TUniformRandomBitGenerator>,
882  "generator must meet the UniformRandomBitGenerator concept");
883 
884  std::uniform_real_distribution<Scalar> uniform(Scalar(0), Scalar(1));
885  std::uniform_real_distribution<Scalar> uniform_twopi(
886  Scalar(0), 2 * kPi<Scalar>);
887

889  const Scalar u2 = uniform_twopi(generator);
890  const Scalar u3 = uniform_twopi(generator);
891 
892  const Scalar a = sqrt(1 - u1);
893  const Scalar b = sqrt(u1);
894 
895  return So3(
896  QuaternionMember(a * sin(u2), a * cos(u2), b * sin(u3), b * cos(u3)));
