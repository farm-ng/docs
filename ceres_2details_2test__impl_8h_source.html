103  EIGEN_MAKE_ALIGNED_OPERATOR_NEW

107  template <class TScalar>
108  bool operator()(TScalar const* const s_t_wa, TScalar* s_residuals) const {
109  Eigen::Map<LieGroup<TScalar> const> const t_wa(s_t_wa);
110  // Mapper class is only used to facciliate difference between
111  // So2 (which uses Scalar as tangent vector type) and other groups
112  // (which use Eigen::Vector<...> as tangent vector type).
113  //
114  // Feel free to use direct dereferencing or Eigen::Map depending
115  // on ypur use-case for concrete application
116  //
117  // We only use Mapper class in order to make tests universally
118  // compatible with LieGroup::Tangent being Scalar or Eigen::Vector
119  using Mapper = Mapper<typename LieGroup<TScalar>::Tangent>;
120  typename Mapper::Map residuals = Mapper::map(s_residuals);
121 
122  // We are able to mix Sophus types with doubles and Jet types withou
123  // needing to cast to TScalar.
124  residuals = (t_aw * t_wa).log();
125  // Reverse order of multiplication. This forces the compiler to verify
126  // that (Jet, double) and (double, Jet) LieGroup multiplication work
127  // correctly.
128  residuals = (t_wa * t_aw).log();
129  // Finally, ensure that Jet-to-Jet multiplication works.
130  residuals = (t_wa * t_aw.template cast<TScalar>()).log();
131  return true;
132  }
133 
134  LieGroupF64 t_aw;
135  };

226  bool testAveraging(
227  const size_t num_vertices,
228  double const sigma_init,
229  double const sigma_observation) {
230  if (num_vertices == 0u) {
231  return true;
232  }
233  double const sigma_init_elementwise = sigma_init / std::sqrt(kDoF);
234  double const sigma_observation_elementwise =
235  sigma_observation / std::sqrt(kDoF);
236  // Running Lie group averaging on a K_n graph with a random initialization
237  // noise and random noise in observations
238  ceres::Problem problem;
239 
240  // "Random" initialization in order to keep tests repeatable
241  std::mt19937 rng(2021);
242  StdVector<LieGroupF64> vec(num_vertices);
243 
244  StdVector<LieGroupF64> v_estimate;
245  v_estimate.reserve(num_vertices);
246  double initial_error = 0.;
247  auto parametrization = new sophus::Manifold<TLieGroup>;
248 
249  // All vertices are initialized with an i.i.d noise with normal
250  // distribution; Scaling is adjusted in order to maintain the same
251  // expectation of squared norm for all groups
252  for (size_t i = 0; i < num_vertices; ++i) {
253  auto& v = vec[i];
254  v = LieGroupF64::sampleUniform(rng);
255  const Tangentd delta_log =
256  Random<Tangentd>::sample(rng) * sigma_init_elementwise;
257  const LieGroupF64 delta = LieGroupF64::exp(delta_log);
258  v_estimate.emplace_back(v * delta);
259  initial_error += squaredNorm(delta_log);
260  problem.AddParameterBlock(
261  v_estimate.back().data(),
262  LieGroupF64::kNumParameters,
263  parametrization);
264  }
265 
266  // For simplicity of graph generation, we use a complete (undirected) graph.
267  // Each edge (observation) has i.i.d noise with multivariate normal
268  // distribution; Scaling is adjusted in order to maintain the same
269  // expectation of squared norm for all groups
270  for (size_t i = 0; i < num_vertices; ++i) {
271  for (size_t j = i + 1; j < num_vertices; ++j) {
272  LieGroupF64 diff = vec[i].inverse() * vec[j];
273  auto const delta_log =
274  Random<typename LieGroupF64::Tangent>::sample(rng) *
275  sigma_observation_elementwise;
276  auto const delta = LieGroupF64::exp(delta_log);
277  ceres::CostFunction* cost = new ceres::AutoDiffCostFunction<
278  TestGraphFunctor,
279  LieGroupF64::kDoF,
280  LieGroupF64::kNumParameters,
281  LieGroupF64::kNumParameters>(new TestGraphFunctor(diff * delta));
282  // For real-world problems you should consider using robust
283  // loss-function
284  problem.AddResidualBlock(
285  cost, nullptr, v_estimate[i].data(), v_estimate[j].data());
286  }
287  }
288 
289  ceres::Solver::Options options;
290  options.gradient_tolerance = 1e-2 * sophus::kEpsilonF64;
291  options.function_tolerance = 1e-2 * sophus::kEpsilonF64;
292  options.parameter_tolerance = 1e-2 * sophus::kEpsilonF64;
293  options.linear_solver_type = ceres::SPARSE_NORMAL_CHOLESKY;
294 
295  ceres::Solver::Summary summary;
296  Solve(options, &problem, &summary);
297 
298  // Computing final error in the estimates
299  double final_error = 0.;
300  for (size_t i = 0; i < num_vertices; ++i) {
301  final_error += squaredNorm((vec[i].inverse() * v_estimate[i]).log());
302  }
303 
304  // Expecting reasonable decrease of both estimates' errors and residuals
305  return summary.final_cost < .25 * summary.initial_cost &&
306  final_error < .25 * initial_error;
307  }

309  bool test(
310  LieGroupF64 const& t_w_targ,
311  LieGroupF64 const& t_w_init,
312  PointF64 const& point_a_init,
313  PointF64 const& point_b) {
314  static int constexpr kNumPointParameters = PointF64::RowsAtCompileTime;
315 
316  // Optimization parameters.
317  LieGroupF64 t_wr = t_w_init;
318  PointF64 point_a = point_a_init;
319 
320  // Build the problem.
321  ceres::Problem problem;
322 
323  // Specify local update rule for our parameter
324 
325  auto parameterization = new sophus::Manifold<TLieGroup>;
326  problem.AddParameterBlock(t_wr.data(), kNumParameters, parameterization);
327 
328  // Create and add cost functions. Derivatives will be evaluated via
329  // automatic differentiation
330  ceres::CostFunction* cost_function1 = new ceres::AutoDiffCostFunction<
331  TestLieGroupCostFunctor,
332  LieGroupF64::kDoF,
333  LieGroupF64::kNumParameters>(
334  new TestLieGroupCostFunctor(t_w_targ.inverse()));
335  problem.AddResidualBlock(cost_function1, nullptr, t_wr.data());
336  ceres::CostFunction* cost_function2 = new ceres::AutoDiffCostFunction<
337  TestPointCostFunctor,
338  kNumPointParameters,
339  kNumParameters,
340  kNumPointParameters>(
341  new TestPointCostFunctor(t_w_targ.inverse(), point_b));
342  problem.AddResidualBlock(
343  cost_function2, nullptr, t_wr.data(), point_a.data());
344 
345  // Set solver options (precision / method)
346  ceres::Solver::Options options;
347  options.gradient_tolerance = 0.01 * sophus::kEpsilonF64;
348  options.function_tolerance = 0.01 * sophus::kEpsilonF64;
349  options.linear_solver_type = ceres::DENSE_QR;
350  options.max_num_iterations = 100;
351 
352  // Solve
353  ceres::Solver::Summary summary;
354  Solve(options, &problem, &summary);
355 
356  // Difference between target and parameter
357  double const mse = squaredNorm((t_w_targ.inverse() * t_wr).log());
358  bool const passed = mse < 10. * sophus::kEpsilonF64;
359  return passed;
360  }
