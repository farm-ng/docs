45  static auto areParamsValid(
46  Eigen::Vector<Scalar, kNumParams> const& non_zero_complex)
47  -> sophus::Expected<Success> {
48  static const Scalar kThr = kEpsilon<Scalar> * kEpsilon<Scalar>;
49  const Scalar squared_norm = non_zero_complex.squaredNorm();
50  using std::abs;
51  if (!(squared_norm > kThr || squared_norm < 1.0 / kThr)) {
52  return SOPHUS_UNEXPECTED(
53  "complex number ({}, {}) is too large or too small.\n"
54  "Squared norm: {}, thr: {}",
55  non_zero_complex[0],
56  non_zero_complex[1],
57  squared_norm,
58  kThr);
59  }
60  return sophus::Expected<Success>{};
61  }

65  static auto exp(Eigen::Vector<Scalar, kDof> const& angle_logscale)
66  -> Eigen::Vector<Scalar, kNumParams> {
67  using std::exp;
68  using std::max;
69  using std::min;
70 
71  Scalar const sigma = angle_logscale[1];
72  Scalar s = exp(sigma);
73  // Ensuring proper scale
74  s = max(s, kEpsilonPlus<Scalar>);
75  s = min(s, Scalar(1.) / kEpsilonPlus<Scalar>);
76  Eigen::Vector2<Scalar> z =
77  Rotation2Impl<Scalar>::exp(angle_logscale.template head<1>());
78  z *= s;
79  return z;
80  }

87  theta_sigma[1] = log(complex.norm());
88  return theta_sigma;

110  static auto inverse(Eigen::Vector<Scalar, kNumParams> const& non_zero_complex)
111  -> Eigen::Vector<Scalar, kNumParams> {
112  Scalar squared_scale = non_zero_complex.squaredNorm();
113  return Eigen::Vector<Scalar, kNumParams>(
114  non_zero_complex.x() / squared_scale,
115  -non_zero_complex.y() / squared_scale);
116  }

118  static auto multiplication(
119  Eigen::Vector<Scalar, kNumParams> const& lhs_params,
120  Eigen::Vector<Scalar, kNumParams> const& rhs_params)
121  -> Eigen::Vector<Scalar, kNumParams> {
122  auto result = Complex::multiplication(lhs_params, rhs_params);
123  Scalar const squared_scale = result.squaredNorm();
124 
125  if (squared_scale < kEpsilon<Scalar> * kEpsilon<Scalar>) {
126  /// Saturation to ensure class invariant.
127  result.normalize();
128  result *= kEpsilonPlus<Scalar>;
129  }
130  if (squared_scale > Scalar(1.) / (kEpsilon<Scalar> * kEpsilon<Scalar>)) {
131  /// Saturation to ensure class invariant.
132  result.normalize();
133  result /= kEpsilonPlus<Scalar>;
134  }
135  return result;
136  }

207  static auto dxExpX(Eigen::Vector<Scalar, kDof> const& a)
208  -> Eigen::Matrix<Scalar, kNumParams, kDof> {
209  using std::cos;
210  using std::exp;
211  using std::sin;
212  Scalar const theta = a[0];
213  Scalar const sigma = a[1];
214 
215  Eigen::Matrix<Scalar, 2, 2> d;
216  // clang-format off
217  d << -sin(theta), cos(theta),
218  cos(theta), sin(theta);
219  // clang-format on
220  return d * exp(sigma);
221  }

223  static auto dxExpXAt0() -> Eigen::Matrix<Scalar, kNumParams, kDof> {
224  Eigen::Matrix<Scalar, 2, 2> d;
225  Scalar const i(1.);

228  d << o, i,

232  }
