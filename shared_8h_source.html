24 // Represents a Non-nullable pointer with shared ownership
25 // Is essentially an adapter between std::shared_ptr and farm_ng::Expected
26 template <class T>
27 class Shared {
28  public:
29  using ExpectedT = farm_ng::Expected<Shared<T>>;
30  using BadExpectedAccess = tl::bad_expected_access<farm_ng::Error>;
31 
32  /// Construct from a possibly null shared_ptr
33  /// The return value is an object containing either a non-null Shared object
34  /// pointer,
35  // or an farm_ng::Error object
36  static ExpectedT tryFrom(std::shared_ptr<T> const& maybe_null) noexcept {
37  if (!maybe_null) {
38  return FARM_UNEXPECTED("is null");
39  }
40  // For some reason, Expected seems to have trouble accepting an r-value
41  // Shared<T> with its deleted copy constructor.
42  Shared<T> temp(maybe_null);
43  return temp;
44  }
45 
46  /// Construct and also makes interior object T
47  /// The return value is an object containing either a non-null Shared object
48  /// pointer, for the new object, or an farm_ng::Error object if memory
49  /// allocation failst or the constructor for the object throws
50  template <class... Args>
51  static ExpectedT tryMake(Args&&... args) noexcept {
52  try {
53  // can throw on bad memory allocation and exceptions in constructor of T.
54  return tryFrom(std::make_shared<T>(std::forward<Args>(args)...));
55  } catch (std::exception& e) {
56  return FARM_UNEXPECTED(e.what());
57  }
58  return FARM_UNEXPECTED();
59  }
60 
61  /// Construct from a possibly null shared_ptr
62  /// Panics if shared is null. See `tryFrom()` for alternate.
63  static Shared from(std::shared_ptr<T> const& shared) {
64  auto maybe = tryFrom(shared);
65  return FARM_UNWRAP(maybe);
66  }
67 
68  /// Construct and also makes interior object T
69  /// Panics if the constructor throws. See `tryFrom()` for alternate.
70  template <class... Args>
71  static Shared<T> make(Args&&... args) {
72  auto maybe = tryMake(std::forward<Args>(args)...);
73  return FARM_UNWRAP(maybe);
74  }
75 
76  /// Returns the interior object which is guarenteed to be available
77  T& operator*() { return *non_null_shared_; }
78 
79  /// Returns the interior object which is guarenteed to be available
80  T const& operator*() const { return *non_null_shared_; }
81 
82  /// Returns the interior object which is guarenteed to be available
83  T* operator->() { return non_null_shared_.get(); }
84 
85  /// Returns the interior object which is guarenteed to be available
86  T const* operator->() const { return non_null_shared_.get(); }
87 
88  // Implicit conversion to a nullable std::shared_ptr okay
89  operator std::shared_ptr<T>() const { return sharedPtr(); }
90 
91  // Return a nullable shared_ptr<T> from this Shared<T> object
92  std::shared_ptr<T> sharedPtr() const { return non_null_shared_; }
93 
94  // Return the raw point from this Shared<T> object
95  T const* ptr() const { return sharedPtr().get(); }
96 
97  // Return the raw point from this Shared<T> object
98  T* ptr() { return sharedPtr().get(); }
99 
100  // Default constructable only if T is
101  Shared() requires(std::is_default_constructible<T>::value)
102  : non_null_shared_(std::make_shared<T>()) {
103  checkMaybeThrow();
104  }
105 
106  // Copy constructor from derived bases
107  template <DerivedFrom<T> Derived>
108  Shared(Shared<Derived> const& other) : non_null_shared_(other.sharedPtr()) {
109  checkMaybeThrow();
110  }
111 
112  // Construct from shared_ptr
113  template <DerivedFrom<T> Derived>
114  Shared(std::shared_ptr<Derived> const& panic_if_null)
115  : non_null_shared_(panic_if_null) {
116  checkMaybeThrow();
117  }
118 
119  // Take ownership from unique_ptr
120  template <DerivedFrom<T> Derived>
121  Shared(std::unique_ptr<Derived>&& panic_if_null)
122  : non_null_shared_(std::move(panic_if_null)) {
123  checkMaybeThrow();
124  }
125 
126  // Not sure why this is needed when the generic one
127  // is defined above
128  Shared(std::shared_ptr<T> const& panic_if_null)
129  : non_null_shared_(panic_if_null) {
130  checkMaybeThrow();
131  }
132 
133  Shared(Shared<T>&& o) : non_null_shared_(o.non_null_shared_) {
134  // We mustn't move the internal shared_ptr
135  // because that would break the invariant.
136  }
137 
138  Shared<T>& operator=(Shared<T>&& o) {
139  // We maintain the invariant since
140  // o.non_null_shared_ must also be valid
141  non_null_shared_ = o.non_null_shared_;
142  return *this;
143  }
144 
145  Shared(Shared<T> const&) = default;
146  Shared<T>& operator=(Shared<T> const&) = default;
147 
148  bool operator==(Shared<T> const& rhs) const noexcept {
149  return this->non_null_shared_ == rhs.non_null_shared_;
150  }
151 
152  private:
153  void checkMaybeThrow() const {
154  if (!non_null_shared_) {
155  throw BadExpectedAccess(FARM_ERROR_REPORT("Shared is null."));
156  }
157  }
158 
159  // Class invariant:non_null_shared_ is guaranteed not to be null.
160  std::shared_ptr<T> non_null_shared_;
161 };
