37 template <class TScalar, int kDim>
38 class UnitVectorImpl {
39  public:
40  using Scalar = TScalar;
41 
42  static int constexpr kDof = kDim - 1;
43  static int constexpr kNumParams = kDim;
44 
45  static auto areParamsValid(
46  Eigen::Vector<Scalar, kNumParams> const& unit_vector)
47  -> sophus::Expected<Success> {
48  static const Scalar kThr = kEpsilon<Scalar>;
49  const Scalar squared_norm = unit_vector.squaredNorm();
50  using std::abs;
51  if (!(abs(squared_norm - 1.0) <= kThr)) {
52  return SOPHUS_UNEXPECTED(
53  "unit vector ({}) is not of unit length.\n"
54  "Squared norm: {}, thr: {}",
55  unit_vector.transpose(),
56  squared_norm,
57  kThr);
58  }
59  return sophus::Expected<Success>{};
60  }
61 
62  static auto oplus(
63  Eigen::Vector<Scalar, kNumParams> const& params,
64  Eigen::Vector<Scalar, kDof> const& delta)
65  -> Eigen::Vector<Scalar, kNumParams> {
66  return matRx(params) * exp(delta);
67  }
68 
69  static auto ominus(
70  Eigen::Vector<Scalar, kNumParams> const& lhs_params,
71  Eigen::Vector<Scalar, kNumParams> const& rhs_params)
72  -> Eigen::Vector<Scalar, kDof> {
73  return log((matRx(lhs_params).transpose() * rhs_params).eval());
74  }
75 
76  static auto paramsExamples()
77  -> std::vector<Eigen::Vector<Scalar, kNumParams>> {
78  return std::vector<Eigen::Vector<Scalar, kNumParams>>(
79  {Eigen::Vector<Scalar, kNumParams>::UnitX(),
80  Eigen::Vector<Scalar, kNumParams>::UnitY(),
81  -Eigen::Vector<Scalar, kNumParams>::UnitX(),
82  -Eigen::Vector<Scalar, kNumParams>::UnitY()});
83  }

92  static auto tangentExamples() -> std::vector<Eigen::Vector<Scalar, kDof>> {
93  return std::vector<Eigen::Vector<Scalar, kDof>>({
94  Eigen::Vector<Scalar, kDof>::Zero(),
95  0.01 * Eigen::Vector<Scalar, kDof>::UnitX(),
96  0.001 * Eigen::Vector<Scalar, kDof>::Ones(),
97  });
98  }
99 
100  private:
101  static auto matRx(Eigen::Vector<Scalar, kNumParams> const& params)
102  -> Eigen::Matrix<Scalar, kNumParams, kNumParams> {
103  static Eigen::Vector<TScalar, kDim> const kUnitX =
104  Eigen::Vector<TScalar, kDim>::UnitX();
105  if ((kUnitX - params).squaredNorm() < kEpsilon<Scalar>) {
106  return Eigen::Matrix<Scalar, kNumParams, kNumParams>::Identity();
107  }
108  Eigen::Vector<Scalar, kNumParams> v = params - kUnitX;
109  return Eigen::Matrix<Scalar, kNumParams, kNumParams>::Identity() -
110  2.0 * (v * v.transpose()) / v.squaredNorm();
111  }
112 
113  static auto exp(Eigen::Vector<Scalar, kDof> const& delta)
114  -> Eigen::Vector<Scalar, kNumParams> {
115  using std::cos;
116  Eigen::Vector<Scalar, kNumParams> params;
117  Scalar theta = delta.norm();
118  params[0] = cos(theta);
119  params.template tail<kDof>() = sinc(theta) * delta;
120  return params;
121  }
122 
123  static auto log(Eigen::Vector<Scalar, kNumParams> const& params)
124  -> Eigen::Vector<Scalar, kDof> {
125  using std::atan2;
126 
127  static Eigen::Vector<Scalar, kDof> const kUnitX =
128  Eigen::Vector<Scalar, kDof>::UnitX();
129  Scalar x = params[0];
130  Eigen::Vector<Scalar, kDof> tail = params.template tail<kDof>();
131  Scalar theta = tail.norm();
132 
133  if (abs(theta) < kEpsilon<Scalar>) {
134  return atan2(0.0, x) * kUnitX;
135  }
136 
137  return (1.0 / theta) * atan2(theta, x) * tail;
138  }
139 
140  static auto sinc(Scalar x) -> Scalar {
141  using std::abs;
142  using std::sin;
143  if (abs(x) < kEpsilon<Scalar>) {
144  return 1.0 - (1.0 / 6.0) * (x * x);
145  }
146  return sin(x) / x;
147  }
148 };

152 template <class TScalar, int kN>
153 class UnitVector : public linalg::UnitVectorImpl<TScalar, kN> {
154  public:
155  using Scalar = TScalar;
156  using Impl = linalg::UnitVectorImpl<Scalar, kN>;
157  static_assert(concepts::ManifoldImpl<Impl>);
158 
159  static int constexpr kDof = kN - 1;
160  static int constexpr kNumParams = kN;
161 
162  static auto tryFromUnitVector(Eigen::Matrix<TScalar, kN, 1> const& v)
163  -> Expected<UnitVector> {
164  SOPHUS_TRY(auto maybe_valid, Impl::areParamsValid(v));
165  UnitVector unit_vector;
166  unit_vector.vector_ = v;
167  return unit_vector;
168  }
169 
170  static auto fromParams(Eigen::Matrix<TScalar, kN, 1> const& v) -> UnitVector {
171  Expected<UnitVector> e_vec = tryFromUnitVector(v);
172  if (!e_vec.has_value()) {
173  SOPHUS_PANIC("{}", e_vec.error());
174  }
175  return e_vec.value();
176  }
177 
178  // Precondition: v must be of unit length.
179  static auto fromUnitVector(Eigen::Matrix<TScalar, kN, 1> const& v)
180  -> UnitVector {
181  return fromParams(v);
182  }
183 
184  static auto fromVectorAndNormalize(Eigen::Matrix<TScalar, kN, 1> const& v)
185  -> UnitVector {
186  return fromUnitVector(v.normalized());
187  }
188 
189  auto oplus(Eigen::Vector<Scalar, kDof> const& delta) -> UnitVector {
190  UnitVector v;
191  v.vector_ = Impl::oplus(vector_, delta);
192  return v;
193  }
194 
195  auto ominus(UnitVector const& rhs_params) -> Eigen::Vector<Scalar, kDof> {
196  return Impl::ominus(vector_, rhs_params.vector_);
197  }
198 
199  void setParams(Eigen::Matrix<TScalar, kN, 1> const& v) const {
200  vector_.params = v;
201  }
202 
203  [[nodiscard]] auto params() const -> Eigen::Matrix<TScalar, kN, 1> const& {
204  return vector_;
205  }
206 
207  [[nodiscard]] auto unsafeMutPtr() { return this->vector_.data(); }
208  [[nodiscard]] auto ptr() const { return this->vector_.data(); }
209 
210  [[nodiscard]] auto vector() const -> Eigen::Matrix<TScalar, kN, 1> const& {
211  return vector_;
212  }
213 
214  UnitVector(UnitVector const&) = default;
215  auto operator=(UnitVector const&) -> UnitVector& = default;
216 
217  private:
218  UnitVector() {}
219 
220  // Class invariant: v_ is of unit length.
221  Eigen::Matrix<TScalar, kN, 1> vector_;
222 };
