56 /// Sim3 base type - implements Sim3 class but is storage agnostic.
57 ///
58 /// Sim(3) is the group of rotations and translation and scaling in 3d. It is
59 /// the semi-direct product of R+xSO(3) and the 3d Euclidean vector space. The
60 /// class is represented using a composition of RxSo3 for scaling plus
61 /// rotation and a 3-vector for translation.
62 ///
63 /// Sim(3) is neither compact, nor a commutative group.
64 ///
65 /// - 4x4 Eigen::Matrix representation:
66 ///
67 /// ```
68 /// | s*R t |
69 /// | o 1 |
70 /// ```
71 ///
72 /// where ``R`` is a 3x3 rotation matrix, ``s`` a positive scale factor,
73 /// ``t`` a translation 3-vector and ``o`` a 3-column vector of zeros.
74 ///
75 /// - Tangent 7-vector: [upsilon, omega, sigma],
76 ///
77 /// where ``upsilon`` is the translational velocity 3-vector and ``omega`` the
78 /// rotational velocity 3-vector, and ``sigma = log(s)``.
79 ///
80 /// - Internal 7-representation: [t0, t1, t2, qi0, qi1, qi2, qr],
81 ///
82 /// with ``t0, t1, t2`` are the translational components, and ``qi0, qi1, q2``
83 /// the imaginary vector part and ``qr1`` the real part of a non-zero
84 /// quaternion. Here the scale ``s`` is equal to the squared norm of the
85 /// quaternion ``s = |q|^2``.
86 ///
87 /// See RxSo3 for more details of the scaling + rotation representation in 3d.
88 ///
89 template <class TDerived>
90 class Sim3Base {
91  public:
92  using Scalar = typename Eigen::internal::traits<TDerived>::Scalar;
93  using TranslationType =
94  typename Eigen::internal::traits<TDerived>::TranslationType;
95  using RxSo3Type = typename Eigen::internal::traits<TDerived>::RxSo3Type;
96  using QuaternionType = typename RxSo3Type::QuaternionType;

99  /// (three for translation, three for rotation and one for scaling).

102  /// translation).

133  /// Adjoint transformation
134  ///
135  /// This function return the adjoint transformation ``Ad`` of the group
136  /// element ``A`` such that for all ``x`` it holds that
137  /// ``hat(Ad_A * x) = A * hat(x) A^{-1}``. See hat-operator below.
138  ///
139  SOPHUS_FUNC [[nodiscard]] Adjoint adj() const {
140  Eigen::Matrix3<Scalar> const r = rxso3().rotationMatrix();
141  Adjoint res;
142  res.setZero();
143  res.template block<3, 3>(0, 0) = rxso3().matrix();
144  res.template block<3, 3>(0, 3) = So3<Scalar>::hat(translation()) * r;
145  res.template block<3, 1>(0, 6) = -translation();
146 
147  res.template block<3, 3>(3, 3) = r;
148 
149  res(6, 6) = Scalar(1);
150  return res;
151  }

169  /// Logarithmic map
170  ///
171  /// Computes the logarithm, the inverse of the group exponential which maps
172  /// element of the group (rigid body transformations) to elements of the
173  /// tangent space (twist).
174  ///
175  /// To be specific, this function computes ``vee(logmat(.))`` with
176  /// ``logmat(.)`` being the matrix logarithm and ``vee(.)`` the vee-operator
177  /// of Sim(3).
178  ///
179  SOPHUS_FUNC [[nodiscard]] Tangent log() const {
180  // The derivation of the closed-form Sim(3) logarithm for is done
181  // analogously to the closed-form solution of the SE(3) logarithm, see
182  // J. Gallier, D. Xu, "Computing exponentials of skew symmetric matrices
183  // and logarithms of orthogonal matrices", IJRA 2002.
184  // https:///pdfs.semanticscholar.org/cfe3/e4b39de63c8cabd89bf3feff7f5449fc981d.pdf
185  // (Sec. 6., pp. 8)
186  Tangent res;
187  auto omega_sigma_and_theta = rxso3().logAndTheta();
188  Eigen::Vector3<Scalar> const omega =
189  omega_sigma_and_theta.tangent.template head<3>();
190  Scalar sigma = omega_sigma_and_theta.tangent[3];
191  Eigen::Matrix3<Scalar> const mat_omega = So3<Scalar>::hat(omega);
192  Eigen::Matrix3<Scalar> const w_inv = details::calcWInv<Scalar, 3>(
193  mat_omega, omega_sigma_and_theta.theta, sigma, scale());
194 
195  res.segment(0, 3) = w_inv * translation();
196  res.segment(3, 3) = omega;
197  res[6] = sigma;
198  return res;
199  }

201  /// Returns 4x4 matrix representation of the instance.
202  ///
203  /// It has the following form:
204  ///
205  /// | s*R t |
206  /// | o 1 |
207  ///
208  /// where ``R`` is a 3x3 rotation matrix, ``s`` a scale factor, ``t`` a
209  /// translation 3-vector and ``o`` a 3-column vector of zeros.
210  ///

212  Transformation homogenious_matrix;
213  homogenious_matrix.template topLeftCorner<3, 4>() = matrix3x4();
214  homogenious_matrix.row(3) =
215  Eigen::Matrix<Scalar, 4, 1>(Scalar(0), Scalar(0), Scalar(0), Scalar(1));
216  return homogenious_matrix;
217  }

250  /// Group action on 3-points.
251  ///
252  /// This function rotates, scales and translates a three dimensional point
253  /// ``p`` by the Sim(3) element ``(bar_sR_foo, t_bar)`` (= similarity
254  /// transformation):
255  ///
256  /// ``p_bar = bar_sR_foo * p_foo + t_bar``.
257  ///
258  template <
259  typename TPointDerived,
260  typename = typename std::enable_if<
261  IsFixedSizeVector<TPointDerived, 3>::value>::type>
262  SOPHUS_FUNC PointProduct<TPointDerived> operator*(
263  Eigen::MatrixBase<TPointDerived> const& p) const {
264  return rxso3() * p + translation();
265  }
266 
267  /// Group action on homogeneous 3-points. See above for more details.
268  ///
269  template <
270  typename THPointDerived,
271  typename = typename std::enable_if<
272  IsFixedSizeVector<THPointDerived, 4>::value>::type>
273  SOPHUS_FUNC HomogeneousPointProduct<THPointDerived> operator*(
274  Eigen::MatrixBase<THPointDerived> const& p) const {
275  PointProduct<THPointDerived> const tp =
276  rxso3() * p.template head<3>() + p(3) * translation();
277  return HomogeneousPointProduct<THPointDerived>(tp(0), tp(1), tp(2), p(3));
278  }

280  /// Group action on lines.
281  ///
282  /// This function rotates, scales and translates a parametrized line
283  /// ``l(t) = o + t * d`` by the Sim(3) element:
284  ///
285  /// Origin ``o`` is rotated, scaled and translated
286  /// Direction ``d`` is rotated
287  ///
288  SOPHUS_FUNC Line operator*(Line const& l) const {
289  Line rotated_line = rxso3() * l;
290  return Line(
291  rotated_line.origin() + translation(), rotated_line.direction());
292  }
293 
294  /// Group action on planes.
295  ///
296  /// This function rotates and translates a plane
297  /// ``n.x + d = 0`` by the Sim(3) element:
298  ///
299  /// Normal vector ``n`` is rotated
300  /// Offset ``d`` is adjusted for scale and translation
301  ///
302  SOPHUS_FUNC Hyperplane operator*(Hyperplane const& p) const {
303  Hyperplane const rotated = rxso3() * p;
304  return Hyperplane(
305  rotated.normal(),
306  rotated.offset() - translation().dot(rotated.normal()));
307  }

572  /// Returns derivative of exp(x) wrt. x.
573  ///
574  SOPHUS_FUNC static Eigen::Matrix<Scalar, kNumParameters, kDoF> dxExpX(
575  Tangent const& vec_a) {
576  Eigen::Matrix<Scalar, kNumParameters, kDoF> dx;
577 
578  static Eigen::Matrix3<Scalar> const kI = Eigen::Matrix3<Scalar>::Identity();
579  Eigen::Vector3<Scalar> const vec_omega = vec_a.template segment<3>(3);
580  Eigen::Vector3<Scalar> const vec_upsilon = vec_a.template head<3>();
581  Scalar const sigma = vec_a[6];
582  Scalar const theta = vec_omega.norm();
583 
584  Eigen::Matrix3<Scalar> const mat_omega = So3<Scalar>::hat(vec_omega);
585  Eigen::Matrix3<Scalar> const mat_omega2 = mat_omega * mat_omega;
586  Eigen::Vector3<Scalar> theta_domega;
587  if (theta < kEpsilon<Scalar>) {
588  theta_domega = Eigen::Vector3<Scalar>::Zero();
589  } else {
590  theta_domega = vec_omega / theta;
591  }
592  static Eigen::Matrix3<Scalar> const kOmegaDomega[3] = {
593  So3<Scalar>::hat(Eigen::Vector3<Scalar>::Unit(0)),
594  So3<Scalar>::hat(Eigen::Vector3<Scalar>::Unit(1)),
595  So3<Scalar>::hat(Eigen::Vector3<Scalar>::Unit(2))};
596 
597  Eigen::Matrix3<Scalar> const omega2_domega[3] = {
598  kOmegaDomega[0] * mat_omega + mat_omega * kOmegaDomega[0],
599  kOmegaDomega[1] * mat_omega + mat_omega * kOmegaDomega[1],
600  kOmegaDomega[2] * mat_omega + mat_omega * kOmegaDomega[2]};
601 
602  Eigen::Matrix3<Scalar> const w =
603  details::calcW<Scalar, 3>(mat_omega, theta, sigma);
604 
605  dx.template block<4, 3>(0, 0).setZero();
606  dx.template block<4, 4>(0, 3) =
607  RxSo3<Scalar>::dxExpX(vec_a.template tail<4>());
608  dx.template block<3, 4>(4, 3).setZero();

662  /// Group exponential
663  ///
664  /// This functions takes in an element of tangent space and returns the
665  /// corresponding element of the group Sim(3).
666  ///
667  /// The first three components of ``a`` represent the translational part
668  /// ``upsilon`` in the tangent space of Sim(3), the following three components
669  /// of ``a`` represents the rotation vector ``omega`` and the final component
670  /// represents the logarithm of the scaling factor ``sigma``.
671  /// To be more specific, this function computes ``expmat(hat(a))`` with
672  /// ``expmat(.)`` being the matrix exponential and ``hat(.)`` the hat-operator
673  /// of Sim(3), see below.
674  ///
675  SOPHUS_FUNC static Sim3<Scalar> exp(Tangent const& vec_a) {
676  // For the derivation of the exponential map of Sim(3) see
677  // H. Strasdat, "Local Accuracy and Global Consistency for Efficient Visual
678  // SLAM", PhD thesis, 2012.
679  // http:///hauke.strasdat.net/files/strasdat_thesis_2012.pdf (A.5, pp. 186)
680  Eigen::Vector3<Scalar> const vec_upsilon = vec_a.segment(0, 3);
681  Eigen::Vector3<Scalar> const vec_omega = vec_a.segment(3, 3);
682  Scalar const sigma = vec_a[6];
683  auto rxso3_and_theta = RxSo3<Scalar>::expAndTheta(vec_a.template tail<4>());
684  Eigen::Matrix3<Scalar> const mat_omega = So3<Scalar>::hat(vec_omega);
685 
686  Eigen::Matrix3<Scalar> const w =
687  details::calcW<Scalar, 3>(mat_omega, rxso3_and_theta.theta, sigma);
688  return Sim3<Scalar>(rxso3_and_theta.rxso3, w * vec_upsilon);
689  }
690 
691  /// Returns the ith infinitesimal generators of Sim(3).
692  ///
693  /// The infinitesimal generators of Sim(3) are:
694  ///
695  /// ```
696  /// | 0 0 0 1 |
697  /// G_0 = | 0 0 0 0 |
698  /// | 0 0 0 0 |
699  /// | 0 0 0 0 |
700  ///
701  /// | 0 0 0 0 |
702  /// G_1 = | 0 0 0 1 |
703  /// | 0 0 0 0 |
704  /// | 0 0 0 0 |
705  ///
706  /// | 0 0 0 0 |
707  /// G_2 = | 0 0 0 0 |
708  /// | 0 0 0 1 |
709  /// | 0 0 0 0 |
710  ///
711  /// | 0 0 0 0 |
712  /// G_3 = | 0 0 -1 0 |
713  /// | 0 1 0 0 |
714  /// | 0 0 0 0 |
715  ///
716  /// | 0 0 1 0 |
717  /// G_4 = | 0 0 0 0 |
718  /// | -1 0 0 0 |
719  /// | 0 0 0 0 |
720  ///
721  /// | 0 -1 0 0 |
722  /// G_5 = | 1 0 0 0 |
723  /// | 0 0 0 0 |
724  /// | 0 0 0 0 |
725  ///
726  /// | 1 0 0 0 |
727  /// G_6 = | 0 1 0 0 |
728  /// | 0 0 1 0 |
729  /// | 0 0 0 0 |
730  /// ```
731  ///
732  /// Precondition: ``i`` must be in [0, 6].
733  ///
734  SOPHUS_FUNC static Transformation generator(int i) {
735  SOPHUS_ASSERT(i >= 0 || i <= 6, "i should be in range [0,6].");
736  Tangent e;
737  e.setZero();
738  e[i] = Scalar(1);
739  return hat(e);
740  }
741 
742  /// hat-operator
743  ///
744  /// It takes in the 7-vector representation and returns the corresponding
745  /// matrix representation of Lie algebra element.
746  ///
747  /// Formally, the hat()-operator of Sim(3) is defined as
748  ///
749  /// ``hat(.): R^7 -> R^{4x4}, hat(a) = sum_i a_i * G_i`` (for i=0,...,6)
750  ///
751  /// with ``G_i`` being the ith infinitesimal generator of Sim(3).
752  ///
753  /// The corresponding inverse is the vee()-operator, see below.
754  ///
755  SOPHUS_FUNC static Transformation hat(Tangent const& vec_a) {
756  Transformation vec_omega;
757  vec_omega.template topLeftCorner<3, 3>() =
758  RxSo3<Scalar>::hat(vec_a.template tail<4>());
759  vec_omega.col(3).template head<3>() = vec_a.template head<3>();
760  vec_omega.row(3).setZero();
761  return vec_omega;
762  }
763 
764  /// Lie bracket
765  ///
766  /// It computes the Lie bracket of Sim(3). To be more specific, it computes
767  ///
768  /// ``[omega_1, omega_2]_sim3 := vee([hat(omega_1), hat(omega_2)])``
769  ///
770  /// with ``[A,B] := AB-BA`` being the matrix commutator, ``hat(.)`` the
771  /// hat()-operator and ``vee(.)`` the vee()-operator of Sim(3).
772  ///
773  SOPHUS_FUNC static Tangent lieBracket(
774  Tangent const& vec_a, Tangent const& vec_b) {
775  Eigen::Vector3<Scalar> const vec_upsilon1 = vec_a.template head<3>();
776  Eigen::Vector3<Scalar> const vec_upsilon2 = vec_b.template head<3>();
777  Eigen::Vector3<Scalar> const vec_omega1 = vec_a.template segment<3>(3);
778  Eigen::Vector3<Scalar> const vec_omega2 = vec_b.template segment<3>(3);
779  Scalar sigma1 = vec_a[6];
780  Scalar sigma2 = vec_b[6];
781 
782  Tangent res;
783  res.template head<3>() = So3<Scalar>::hat(vec_omega1) * vec_upsilon2 +
784  So3<Scalar>::hat(vec_upsilon1) * vec_omega2 +
785  sigma1 * vec_upsilon2 - sigma2 * vec_upsilon1;
786  res.template segment<3>(3) = vec_omega1.cross(vec_omega2);
787  res[6] = Scalar(0);
788 
789  return res;
790  }
791 
792  /// Draw uniform sample from Sim(3) manifold.
793  ///
794  /// Translations are drawn component-wise from the range [-1, 1].
795  /// The scale factor is drawn uniformly in log2-space from [-1, 1],
796  /// hence the scale is in [0.5, 2].
797  ///
798  template <class TUniformRandomBitGenerator>
799  static Sim3 sampleUniform(TUniformRandomBitGenerator& generator) {
800  std::uniform_real_distribution<Scalar> uniform(Scalar(-1), Scalar(1));
801  return Sim3(
802  RxSo3<Scalar>::sampleUniform(generator),
803  Eigen::Vector3<Scalar>(
804  uniform(generator), uniform(generator), uniform(generator)));
805  }
806 
807  /// vee-operator
808  ///
809  /// It takes the 4x4-matrix representation ``Omega`` and maps it to the
810  /// corresponding 7-vector representation of Lie algebra.
811  ///
812  /// This is the inverse of the hat()-operator, see above.
813  ///
814  /// Precondition: ``Omega`` must have the following structure:
815  ///
816  /// | g -f e a |
817  /// | f g -d b |
818  /// | -e d g c |
819  /// | 0 0 0 0 |
820  ///
821  SOPHUS_FUNC static Tangent vee(Transformation const& mat_omega) {
822  Tangent vec_upsilon_omega_sigma;
823  vec_upsilon_omega_sigma.template head<3>() =
824  mat_omega.col(3).template head<3>();
825  vec_upsilon_omega_sigma.template tail<4>() =
826  RxSo3<Scalar>::vee(mat_omega.template topLeftCorner<3, 3>());
827  return vec_upsilon_omega_sigma;
828  }
829 
830  protected:
831  RxSo3Member rxso3_; // NOLINT
832  TranslationMember translation_; // NOLINT
833 };
