20 template <class TT>
21 Eigen::Matrix<TT, 3, 3> nonOrthogonalityMatrix(
22  Eigen::Matrix<TT, 3, 1> const& non_orthogonality) {
23  Eigen::Matrix<TT, 3, 3> lower_triagonal = Eigen::Matrix<TT, 3, 3>::Identity();
24  lower_triagonal(1, 0) = non_orthogonality[0];
25  lower_triagonal(2, 0) = non_orthogonality[1];
26  lower_triagonal(2, 1) = non_orthogonality[2];
27  return lower_triagonal;
28 }

33  ScalingNonOrthogonalityGyroModel(
34  Eigen::Matrix<TT, 3, 1> const& scale = Eigen::Matrix<TT, 3, 1>::Ones(),
35  Eigen::Matrix<TT, 3, 1> const& non_orthogonality =
36  Eigen::Matrix<TT, 3, 1>::Zero(),
37  Eigen::Matrix<TT, 3, 1> const& gyro_bias =
38  Eigen::Matrix<TT, 3, 1>::Zero())
39  : scale(scale),
40  non_orthogonality(non_orthogonality),
41  gyro_bias(gyro_bias) {}
42 
43  static ScalingNonOrthogonalityGyroModel<TT> fromParams(
44  Eigen::Matrix<TT, 9, 1> const& params) {
45  return ScalingNonOrthogonalityGyroModel<TT>(
46  {params[0], params[1], params[2]},
47  {params[3], params[4], params[5]},
48  {params[6], params[7], params[8]});
49  }
50 
51  [[nodiscard]] Eigen::Matrix<TT, 3, 1> gyroMeasurement(
52  Eigen::Matrix<TT, 3, 1> const& imu_angular_rate_imu) const {
53  return Eigen::DiagonalMatrix<TT, 3>(scale) *
54  nonOrthogonalityMatrix(non_orthogonality) *
55  imu_angular_rate_imu +
56  gyro_bias;
57  }
58 
59  [[nodiscard]] Eigen::Matrix<TT, 9, 1> params() const {
60  Eigen::Matrix<TT, 9, 1> params;
61  params << scale[0], scale[1], scale[2], non_orthogonality[0],
62  non_orthogonality[1], non_orthogonality[2], gyro_bias[0], gyro_bias[1],
63  gyro_bias[2];
64  return params;
65  }
66 
67  Eigen::Matrix<TT, 3, 1> scale;
68  Eigen::Matrix<TT, 3, 1> non_orthogonality;
69  Eigen::Matrix<TT, 3, 1> gyro_bias;
70 };
