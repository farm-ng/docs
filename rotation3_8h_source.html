89 /// Construct rotation which would take unit direction vector ``from`` into
90 /// ``to`` such that ``to = rotThroughPoints(from,to) * from``. I.e. that the
91 /// rotated point ``from`` is colinear with ``to`` (equal up to scale)
92 ///
93 /// The axis of rotation is perpendicular to both ``from`` and ``to``.
94 ///
95 template <class TScalar>
96 auto rotThroughPoints(
97  UnitVector3<TScalar> const& from, UnitVector3<TScalar> const& to)
98  -> Rotation3<TScalar> {
99  using std::abs;
100  using std::atan2;
101  Eigen::Vector<TScalar, 3> from_cross_to = from.vector().cross(to.vector());
102  TScalar n = from_cross_to.norm();
103  if (abs(n) < sophus::kEpsilon<TScalar>) {
104  return Rotation3<TScalar>();
105  }
106  // https://stackoverflow.com/a/32724066
107  TScalar angle = atan2(n, from.vector().dot(to.vector()));
108 
109  return Rotation3<TScalar>::exp(angle * from_cross_to / n);
110 }

112 /// Construct rotation which would take direction vector ``from`` into ``to``
113 /// such that ``to \propto rotThroughPoints(from,to) * from``. I.e. that the
114 /// rotated point ``from`` is colinear with ``to`` (equal up to scale)
115 ///
116 /// The axis of rotation is perpendicular to both ``from`` and ``to``.
117 ///
118 /// Precondition: Neither ``from`` nor ``to`` must be zero. This is
119 // unchecked.
120 template <class TScalar>
121 auto rotThroughPoints(
122  Eigen::Vector<TScalar, 3> const& from, Eigen::Vector<TScalar, 3> const& to)
123  -> Rotation3<TScalar> {
124  using std::abs;
125  using std::atan2;
126  Eigen::Vector<TScalar, 3> from_cross_to = from.cross(to);
127  TScalar n = from_cross_to.norm();
128  if (abs(n) < sophus::kEpsilon<TScalar>) {
129  return Rotation3<TScalar>();
130  }
131  // https://stackoverflow.com/a/32724066
132  TScalar angle = atan2(n, from.dot(to));
133 
134  return Rotation3<TScalar>::exp(angle * from_cross_to / n);
135 }
