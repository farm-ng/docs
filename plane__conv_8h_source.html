30 /// Takes in line normal in reference frame foo and constructs a corresponding
31 /// rotation matrix ``foo_rotation_line``.
32 ///
33 /// Precondition: ``normal_in_foo`` must not be close to zero.
34 ///
35 template <class TScalar>
36 So2<TScalar> so2FromNormal(Eigen::Vector2<TScalar> normal_in_foo) {
37  SOPHUS_ASSERT(
38  normal_in_foo.squaredNorm() > kEpsilon<TScalar>,
39  "{}",
40  normal_in_foo.transpose().eval());
41  normal_in_foo.normalize();
42  return So2<TScalar>(normal_in_foo.y(), -normal_in_foo.x());
43 }

53 /// Takes in plane normal in reference frame foo and constructs a corresponding
54 /// rotation matrix ``foo_rotation_plane``.
55 ///
56 /// Note: The ``plane`` frame is defined as such that the normal points along
57 /// the positive z-axis. One can specify hints for the x-axis and y-axis
58 /// of the ``plane`` frame.
59 ///
60 /// Preconditions:
61 /// - ``normal_in_foo``, ``xDirHint_foo``, ``yDirHint_foo`` must not be close to
62 /// zero.
63 /// - ``xDirHint_foo`` and ``yDirHint_foo`` must be approx. perpendicular.
64 ///
65 template <class TScalar>
66 Eigen::Matrix3<TScalar> rotationFromNormal(
67  Eigen::Vector3<TScalar> const& normal_in_foo,
68  Eigen::Vector3<TScalar> x_dir_hint_foo =
69  Eigen::Vector3<TScalar>(TScalar(1), TScalar(0), TScalar(0)),
70  Eigen::Vector3<TScalar> y_dir_hint_foo =
71  Eigen::Vector3<TScalar>(TScalar(0), TScalar(1), TScalar(0))) {
72  SOPHUS_ASSERT(
73  x_dir_hint_foo.dot(y_dir_hint_foo) < kEpsilon<TScalar>,
74  "xDirHint ({}) and yDirHint ({}) must be perpendicular.",
75  x_dir_hint_foo.transpose(),
76  y_dir_hint_foo.transpose());
77  using std::abs;
78  using std::sqrt;
79  TScalar const x_dir_hint_foo_sqr_length = x_dir_hint_foo.squaredNorm();
80  TScalar const y_dir_hint_foo_sqr_length = y_dir_hint_foo.squaredNorm();
81  TScalar const normal_foo_sqr_length = normal_in_foo.squaredNorm();
82  SOPHUS_ASSERT(
83  x_dir_hint_foo_sqr_length > kEpsilon<TScalar>,
84  "{}",
85  x_dir_hint_foo.transpose());
86  SOPHUS_ASSERT(
87  y_dir_hint_foo_sqr_length > kEpsilon<TScalar>,
88  "{}",
89  y_dir_hint_foo.transpose());
90  SOPHUS_ASSERT(
91  normal_foo_sqr_length > kEpsilon<TScalar>,
92  "{}",
93  normal_in_foo.transpose());
94 
95  Eigen::Matrix3<TScalar> basis_foo;
96  basis_foo.col(2) = normal_in_foo;
97 
98  if (abs(x_dir_hint_foo_sqr_length - TScalar(1)) > kEpsilon<TScalar>) {
99  x_dir_hint_foo.normalize();
100  }
101  if (abs(y_dir_hint_foo_sqr_length - TScalar(1)) > kEpsilon<TScalar>) {
102  y_dir_hint_foo.normalize();
103  }
104  if (abs(normal_foo_sqr_length - TScalar(1)) > kEpsilon<TScalar>) {
105  basis_foo.col(2).normalize();
106  }
107 
108  TScalar abs_x_dot_z = abs(basis_foo.col(2).dot(x_dir_hint_foo));
109  TScalar abs_y_dot_z = abs(basis_foo.col(2).dot(y_dir_hint_foo));
110  if (abs_x_dot_z < abs_y_dot_z) {
111  // basis_foo.z and xDirHint are far from parallel.
112  basis_foo.col(1) = basis_foo.col(2).cross(x_dir_hint_foo).normalized();
113  basis_foo.col(0) = basis_foo.col(1).cross(basis_foo.col(2));
114  } else {
115  // basis_foo.z and yDirHint are far from parallel.
116  basis_foo.col(0) = y_dir_hint_foo.cross(basis_foo.col(2)).normalized();
117  basis_foo.col(1) = basis_foo.col(2).cross(basis_foo.col(0));
118  }
119  TScalar det = basis_foo.determinant();
120  // sanity check
121  SOPHUS_ASSERT(
122  abs(det - TScalar(1)) < kEpsilon<TScalar>,
123  "Determinant of basis is not 1, but {}. Basis is \n{}\n",
124  det,
125  basis_foo);
126  return basis_foo;
127 }

129 /// Takes in plane normal in reference frame foo and constructs a corresponding
130 /// rotation matrix ``foo_rotation_plane``.
131 ///
132 /// See ``rotationFromNormal`` for details.
133 ///
134 template <class TScalar>
135 So3<TScalar> so3FromPlane(Eigen::Vector3<TScalar> const& normal_in_foo) {
136  return So3<TScalar>(rotationFromNormal(normal_in_foo));
137 }

139 /// Returns a line (wrt. to frame ``foo``), given a pose of the ``line`` in
140 /// reference frame ``foo``.
141 ///
142 /// Note: The plane is defined by X-axis of the ``line`` frame.
143 ///
144 template <class TScalar>
145 Eigen::Hyperplane<TScalar, 2> lineFromSe2(Se2<TScalar> const& foo_from_line) {
146  return Eigen::Hyperplane<TScalar, 2>(
147  normalFromSo2(foo_from_line.so2()), foo_from_line.translation());
148 }

150 /// Returns the pose ``T_foo_line``, given a line in reference frame ``foo``.
151 ///
152 /// Note: The line is defined by X-axis of the frame ``line``.
153 ///
154 template <class TScalar>
155 Se2<TScalar> se2FromLine(Eigen::Hyperplane<TScalar, 2> const& line_in_foo) {
156  TScalar const d = line_in_foo.offset();
157  Eigen::Vector2<TScalar> const n = line_in_foo.normal();
158  So2<TScalar> const foo_rotation_plane = so2FromNormal(n);
159  return Se2<TScalar>(foo_rotation_plane, -d * n);
160 }

162 /// Returns a plane (wrt. to frame ``foo``), given a pose of the ``plane`` in
163 /// reference frame ``foo``.
164 ///
165 /// Note: The plane is defined by XY-plane of the frame ``plane``.
166 ///
167 template <class TScalar>
168 Eigen::Hyperplane<TScalar, 3> planeFromSe3(Se3<TScalar> const& foo_from_plane) {
169  return Eigen::Hyperplane<TScalar, 3>(
170  normalFromSo3(foo_from_plane.so3()), foo_from_plane.translation());
171 }

173 /// Returns the pose ``foo_from_plane``, given a plane in reference frame
174 /// ``foo``.
175 ///
176 /// Note: The plane is defined by XY-plane of the frame ``plane``.
177 ///
178 template <class TScalar>
179 Se3<TScalar> se3FromPlane(Eigen::Hyperplane<TScalar, 3> const& plane_in_foo) {
180  TScalar const d = plane_in_foo.offset();
181  Eigen::Vector3<TScalar> const n = plane_in_foo.normal();
182  So3<TScalar> const foo_rotation_plane = so3FromPlane(n);
183  return Se3<TScalar>(foo_rotation_plane, -d * n);
184 }

186 /// Takes in a hyperplane and returns unique representation by ensuring that the
187 /// ``offset`` is not negative.
188 ///
189 template <class TScalar, int kMatrixDim>
190 Eigen::Hyperplane<TScalar, kMatrixDim> makeHyperplaneUnique(
191  Eigen::Hyperplane<TScalar, kMatrixDim> const& plane) {
192  if (plane.offset() >= 0) {
193  return plane;
194  }
195 
196  return Eigen::Hyperplane<TScalar, kMatrixDim>(
197  -plane.normal(), -plane.offset());
198 }
